73 double GetAngle()
74 {
75 uint16_t raw{};
76 if (!Read16(static_cast<uint8_t>(Reg::kRAW_ANGLE), raw)) {
77 // 読み出し失敗時は直前値を返す
78 return (turn_count_ * kTwoPi) + (last_raw_ * kRadPerLsb);
79 }
80 raw &= 0x0FFF;
81
82 int diff = static_cast<int>(raw) - static_cast<int>(last_raw_);
83 if (diff > 2048) {
84 // 巻き戻り: 逆回転に1周
85 turn_count_ -= 1;
86 } else if (diff < -2048) {
87 // 巻き戻り: 正回転に1周
88 turn_count_ += 1;
89 }
90
91 last_raw_ = raw;
92 return (turn_count_ * kTwoPi) + (raw * kRadPerLsb);

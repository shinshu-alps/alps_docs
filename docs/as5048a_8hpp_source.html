67 double GetAngle()
68 {
69 uint16_t raw{};
70 if (!ReadReg(Reg::kANGLE, raw)) {
71 // 読み出し失敗時は直前値を返す
72 return (turn_count_ * kTwoPi) + (last_raw_ * kRadPerLsb);
73 }
74 raw &= 0x3FFF;
75
76 int diff = static_cast<int>(raw) - static_cast<int>(last_raw_);
77 if (diff > 8192) {
78 turn_count_ -= 1; // 逆回転で一周
79 } else if (diff < -8192) {
80 turn_count_ += 1; // 正回転で一周
81 }
82
83 last_raw_ = raw;
84 return (turn_count_ * kTwoPi) + (raw * kRadPerLsb);

141 static uint16_t AddParity(uint16_t cmd)
142 {
143 uint16_t tmp = cmd;
144 int ones = 0;
145 for (int i = 0; i < 15; i++) {
146 if (tmp & 0x1) ones++;
147 tmp >>= 1;
148 }
149 // 1の数が奇数なら MSB=1、偶数なら MSB=0
150 if (ones % 2 == 1) {
151 cmd |= 0x8000;
152 } else {
153 cmd &= ~0x8000;
154 }
155 return cmd;
156 }

158 bool ReadReg(Reg reg, uint16_t & out_value)
159 {
160 uint16_t tx = static_cast<uint16_t>(reg) & 0x3FFF;
161 tx |= 0x4000; // RD bitをセット
162 tx = AddParity(tx);
163
164 uint8_t write_buf[2] = {static_cast<uint8_t>(tx >> 8), static_cast<uint8_t>(tx & 0xFF)};
165 uint8_t read_buf[2] = {0, 0};
166
167 if (!spi_io_.CallTransfer({write_buf, 2}, {read_buf, 2})) {
168 return false;
169 }
170
171 // 次のトランザクションで結果を取得
172 uint8_t nop_write[2] = {0x00, 0x00};
173 uint8_t nop_read[2] = {0, 0};
174 if (!spi_io_.CallTransfer({nop_write, 2}, {nop_read, 2})) {
175 return false;
176 }
177
178 out_value = (static_cast<uint16_t>(nop_read[0]) << 8) | nop_read[1];
179 return true;
180 }

182 bool WriteReg(Reg reg, uint16_t value)
183 {
184 uint16_t addr = static_cast<uint16_t>(reg) & 0x3FFF;
185 uint16_t tx = addr | 0x0000; // Writeコマンド
186 tx = AddParity(tx);
187
188 uint8_t cmd_buf[2] = {static_cast<uint8_t>(tx >> 8), static_cast<uint8_t>(tx & 0xFF)};
189 uint8_t dummy_buf[2] = {0, 0};
190
191 if (!spi_io_.CallTransfer({cmd_buf, 2}, {dummy_buf, 2})) {
192 return false;
193 }
194
195 uint16_t data = value & 0x3FFF;
196 data = AddParity(data);
197
198 uint8_t data_buf[2] = {static_cast<uint8_t>(data >> 8), static_cast<uint8_t>(data & 0xFF)};
199 return spi_io_.CallTransfer({data_buf, 2}, {dummy_buf, 2});
200 }

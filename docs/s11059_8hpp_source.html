120 bool GetRawData(RawData & output)
121 {
122 uint8_t data[8]; //データを受信
123 static const uint8_t data_pos = kRead; //レジスタの読み初めの場所
124
125 //読み込み先の指定
126 if (i2c_io_.CallWrite(kI2cAddress, etl::span<const uint8_t>{&data_pos, 1}) == false) {
127 return false;
128 }
129 //データの受信
130 if (i2c_io_.CallRead(kI2cAddress, data) == false) {
131 return false;
132 }
133
134 output.red = (data[0] << 8 | data[1]);
135 output.green = (data[2] << 8 | data[3]);
136 output.blue = (data[4] << 8 | data[5]);
137 output.infrared = (data[6] << 8 | data[7]);
138
139 return true;

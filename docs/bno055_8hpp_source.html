240  bool GetAcceleration(Acceleration & output)
241  {
242  uint8_t reg_values[6];
243  if (!ReadMultiRegister(AccelDataXLsbAddr, reg_values)) {
244  return false; // 失敗
245  }
246 
247  int16_t raw_x = (reg_values[1] << 8 | reg_values[0]);
248  int16_t raw_y = (reg_values[3] << 8 | reg_values[2]);
249  int16_t raw_z = (reg_values[5] << 8 | reg_values[4]);
250 
251  output.x = raw_x / kAccelLsb;
252  output.y = raw_y / kAccelLsb;
253  output.z = raw_z / kAccelLsb;
254 
255  return true; // 成功

559  void Initialize()
560  {
561  // 設定モードに入る
562  ChangeOperationMode(OperationMode::kConfig);
563 
564  // 電源モードはノーマル
565  WriteRegister(kPwrModeAddr, kPowerModeNormal);
566 
567  // 軸向きの設定
568  RemapAxis(placement_);
569 
570  // 単位,フォーマットの設定 データシート3.6
571  RegValue unit_format_selection;
572  ReadRegister(kUnitSelAddr, unit_format_selection);
573  unit_format_selection &= 0b11111110; // m/s^2
574  unit_format_selection |= 0b00000100; // rps
575  unit_format_selection |= 0b00001000; // rad
576  unit_format_selection &= 0b11101111; // ℃
577  unit_format_selection &= 0b01111111; // Windows
578  WriteRegister(kUnitSelAddr, unit_format_selection);
579 
580  // 9軸フュージョンモードに入ってセンサ測定を開始する
581  ChangeOperationMode(OperationMode::kNDof);
582  }

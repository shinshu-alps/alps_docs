237  bool GetAcceleration(Acceleration & output)
238  {
239  uint8_t reg_values[6];
240  if (!ReadMultiRegister(AccelDataXLsbAddr, reg_values)) {
241  return false; // 失敗
242  }
243 
244  int16_t raw_x = (reg_values[1] << 8 | reg_values[0]);
245  int16_t raw_y = (reg_values[3] << 8 | reg_values[2]);
246  int16_t raw_z = (reg_values[5] << 8 | reg_values[4]);
247 
248  output.x = raw_x / kAccelLsb;
249  output.y = raw_y / kAccelLsb;
250  output.z = raw_z / kAccelLsb;
251 
252  return true; // 成功
253  }

556  void Initialize()
557  {
558  // 設定モードに入る
559  ChangeOperationMode(OperationMode::kConfig);
560 
561  // 電源モードはノーマル
562  WriteRegister(kPwrModeAddr, kPowerModeNormal);
563 
564  // 軸向きの設定
565  RemapAxis(placement_);

567  // 単位,フォーマットの設定 データシート3.6
568  RegValue unit_format_selection;
569  ReadRegister(kUnitSelAddr, unit_format_selection);
570  unit_format_selection &= 0b11111110; // m/s^2
571  unit_format_selection |= 0b00000100; // rps
572  unit_format_selection |= 0b00001000; // rad
573  unit_format_selection &= 0b11101111; // ℃
574  unit_format_selection &= 0b01111111; // Windows
575  WriteRegister(kUnitSelAddr, unit_format_selection);
576 
577  // 9軸フュージョンモードに入ってセンサ測定を開始する
578  ChangeOperationMode(OperationMode::kNDof);

150 void Calibration()
151 {
152 using namespace alps::mbed::communication::odrive;
153 for (size_t i = 0; i < g_odrives.size(); i++)
154 {
155 g_odrives[i].clearErrors();
156 }
157 ThisThread::sleep_for(100ms);
158 for (size_t i = 0; i < g_odrives.size(); i++)
159 {
160 g_odrives[i].setState(alps::cmn::communication::odrive::ODriveAxisState::AXIS_STATE_FULL_CALIBRATION_SEQUENCE);
161 }
162#ifdef DEBUG
163 for (size_t i = 0; i < g_odrives.size(); i++)
164 {
165 uint32_t error = g_odrv_user_datas[i].last_heartbeat.Axis_Error;
166 uint32_t state = g_odrv_user_datas[i].last_heartbeat.Axis_State;
167 printf("odrive%d: axis state : %ld\n", i, state);
168 printf("odrive%d: error code : %ld\n", i, error);
169 }
170#endif
171 ThisThread::sleep_for(1s);
172 std::array<bool, kMaxOdriveNodeId> finish_array{false};
173 std::array<bool, kMaxOdriveNodeId> error_array{false};
174 while (true)
175 {
176 bool all_finish = true;
177 for (size_t i = 0; i < g_odrives.size(); i++)
178 {
179 if (error_array[i] == true)
180 {
181 break;
182 }
183 uint32_t state = odrive::g_odrv_user_datas[i].last_heartbeat.Axis_State;
184 uint32_t error = odrive::g_odrv_user_datas[i].last_heartbeat.Axis_Error;
185 if (state == alps::cmn::communication::odrive::ODriveAxisState::AXIS_STATE_IDLE &&
186 error == alps::cmn::communication::odrive::ODriveError::ODRIVE_ERROR_NONE)
187 {
188 finish_array[i] = true;
189 }
190 if (error != alps::cmn::communication::odrive::ODriveError::ODRIVE_ERROR_NONE)
191 {
192 finish_array[i] = true;
193 error_array[i] = true;
194 }
195 }
196 for (size_t i = 0; i < g_odrives.size(); i++)
197 {
198 if (finish_array[i] == false)
199 {
200 all_finish = false;
201 break;
202 }
203 }
204 if (all_finish)
205 {
206 bool all_success = true;
207 for (size_t i = 0; i < g_odrives.size(); i++)
208 {
209 if (error_array[i] == true)
210 {
211 all_success = false;
212 printf("odrv%d error!!\n", i);
213 PrintError(odrive::g_odrv_user_datas[i].last_heartbeat.Axis_Error);
214 }
215 }
216 if (all_success)
217 {
218 printf("All calibrations were successful!!\n");
219 }
220 break;
221 }
222 ThisThread::sleep_for(100ms);
223 }
224 ThisThread::sleep_for(100ms);

114  void AnalyzeStaticFriction()
115  {
116  using namespace std::chrono_literals;
117 
118  // モーター軸が静止していることを確認
119  while (IsMotorRotating()) {
120  ALPS_LOG_WARN("Motor is rotating. Please stop the motor.");
121  time::SleepFor(1s);
122  }
123 
124  ALPS_LOG_INFO("Start static friction analysis");
125 
126  // モーター出力を徐々に上げていき，モーター軸が回転し始めるまでの出力を計測

143  void AnalyzeKineticFriction()
144  {
145  using namespace std::chrono_literals;
146 
147  float output = static_friction_;
148 
149  // モーター軸が回転していることを確認
150  while (!IsMotorRotating()) {
151  ALPS_LOG_WARN("Motor is not rotating. Increase the motor output.");
152  output += motor_output_change_step_;

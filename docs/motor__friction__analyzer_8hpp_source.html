99 FrictionResult Analyze(
100 float initial_static_analyze_output = 0.0f,
101 etl::optional<float> initial_kinetic_analyze_output = etl::nullopt)
102 {
103 using namespace std::chrono_literals;
104
105 Reset();
106 AnalyzeStaticFriction(initial_static_analyze_output);
107 if (initial_kinetic_analyze_output) {
108 AnalyzeKineticFriction(initial_kinetic_analyze_output.value());
109 } else {
110 AnalyzeKineticFriction(static_friction_ * 1.2); // 静止摩擦よりやや大きめの初期値を指定
111 }
112 ShowFrictionResult();
113
114 return {static_friction_, kinetic_friction_};
115 }

124 bool IsMotorRotating()

166 void AnalyzeStaticFriction(float initial_output = 0.0f)
167 {
168 using namespace std::chrono_literals;
169
170 // モーター軸が静止していることを確認
171 while (IsMotorRotating()) {
172 ALPS_LOG_WARN("Motor is rotating. Please stop the motor.");
173 Wait(1s);
174 }
175
176 ALPS_LOG_INFO("Start static friction analysis");
177
178 // モーター出力を徐々に上げていき，モーター軸が回転し始めるまでの出力を計測
179 float output = initial_output;
180 while (!IsMotorRotating()) {
181 output += motor_output_change_step_;
182 ALPS_LOG_INFO("\t\toutput: %f", output);
183 motor_.CallDrive(output);
184 Wait(static_analyze_interval_);
185 }
186
187 // 静止摩擦を記録
188 static_friction_ = output;
189 ALPS_LOG_INFO("Static friction: %f", static_friction_);
190 }
191
197 void AnalyzeKineticFriction(float initial_output)
198 {
199 using namespace std::chrono_literals;
200
201 float output = initial_output;
202 motor_.CallDrive(output);
203
204 // 定常状態まで待つ
205 ALPS_LOG_INFO("Waiting for steady state.");
206 this->Wait(1s);
207 constexpr float kIncreaseOutputRatio = 1.2f;
208 while (!IsMotorRotating()) {
209 ALPS_LOG_WARN("Motor is not rotating. Increase the motor output.");
210 output *= kIncreaseOutputRatio;
211 ALPS_LOG_INFO("\t\toutput: %f", output);
212 motor_.CallDrive(output);
213 this->Wait(1s);
214 }
215
216 ALPS_LOG_INFO("Start kinetic friction analysis.");
217
218 // モーター出力を徐々に下げていき，モーター軸が回転しなくなるまでの出力を計測
219 while (IsMotorRotating()) {
220 output -= motor_output_change_step_;
221 ALPS_LOG_INFO("\t\toutput: %f", output);
222 motor_.CallDrive(output);
223 Wait(kinetic_analyze_interval_);
224 }
225
226 // 動摩擦を記録
227 kinetic_friction_ = output;
228 ALPS_LOG_INFO("Kinetic friction: %f", kinetic_friction_);
229 }

234 ALPS_LOG_INFO("Friction analysis result");
235 ALPS_LOG_INFO("\t\tStatic friction: %f", static_friction_);
236 ALPS_LOG_INFO("\t\tKinetic friction: %f", kinetic_friction_);

30 static inline ODriveCanIntf wrap_can_intf(TIntf & intf) \

32 return { \
33 &intf, \

35 return sendMsg(*(TIntf *)intf, id, length, data); \
36 }, \
37 [](void * intf) { pumpEvents(*(TIntf *)intf); }}; \

377 void onReceive(uint32_t id, uint8_t length, const uint8_t * data)
378 {
379// #ifdef DEBUG
380// int byte_index = length - 1;
381// printf("received:\n");
382// printf(" id: 0x");
383// printf("%x\n", id);
384// printf(" data: 0x");
385// while (byte_index >= 0) printf("%x", msg.data[byte_index--]);
386// printf("\n");
387// #endif // DEBUG
388 if (node_id_ != (id >> ODriveCAN::kNodeIdShift)) return;
389 switch (id & ODriveCAN::kCmdIdBits) {
390 case Get_Encoder_Estimates_msg_t::cmd_id: {
391 Get_Encoder_Estimates_msg_t estimates;
392 estimates.decode_buf(data);
393 if (feedback_callback_) feedback_callback_(estimates, feedback_user_data_);
394 break;
395 }
396 case Heartbeat_msg_t::cmd_id: {
397 Heartbeat_msg_t status;
398 status.decode_buf(data);
399 if (axis_state_callback_ != nullptr)
400 axis_state_callback_(status, axis_state_user_data_);
401 else
402 ALPS_LOG_INFO("missing callback");
403 break;
404 }
405 default: {
406 if (requested_msg_id_ == REQUEST_PENDING) return;
407// #ifdef DEBUG
408// printf("waiting for: 0x");
409// printf("%x\n", requested_msg_id_);
410// #endif // DEBUG
411 if ((id & ODriveCAN::kCmdIdBits) != requested_msg_id_) return;
412 memcpy(buffer_, data, length);
413 requested_msg_id_ = REQUEST_PENDING;
414 }
415 }

424 template <typename T>
425 bool request(T & msg, uint16_t timeout_ms = 10)
426 {
427 requested_msg_id_ = msg.cmd_id;
428 can_intf_.sendMsg(
429 (node_id_ << ODriveCAN::kNodeIdShift) | msg.cmd_id,
430 0, // no data
431 nullptr // RTR=1
432 );
433 if (!awaitMsg(timeout_ms)) return false;
434 msg.decode_buf(buffer_);
435 return true;
436 }

462 template <typename T>
463 T getEndpoint(uint16_t endpoint_id, uint16_t timeout_ms = 10)
464 {
465 uint8_t data[8] = {};
466 data[0] = 0; // Opcode read
467
468 // Little-endian endpoint
469 data[1] = (uint8_t)(endpoint_id);
470 data[2] = (uint8_t)(endpoint_id >> 8);
471
472 can_intf_.sendMsg((node_id_ << ODriveCAN::kNodeIdShift) | 0x004, 8, data);
473 if (!awaitMsg(timeout_ms)) return T{};
474
475 T ret{};
476 memcpy(&ret, &buffer_[4], sizeof(T));
477 return ret;
478 }
479
490 template <typename T>
491 bool setEndpoint(uint16_t endpoint_id, T value)
492 {
493 uint8_t data[8] = {};
494 data[0] = 1; // Opcode write
495
496 // Endpoint
497 data[1] = endpoint_id & 0xFF;
498 data[2] = (endpoint_id >> 8) & 0xFF;
499
500 // Value to write
501 memcpy(&data[4], &value, sizeof(T));
502
503 can_intf_.sendMsg((node_id_ << ODriveCAN::kNodeIdShift) | 0x004, 8, data);
504 return true;
505 }
506
507private:
508 bool awaitMsg(uint16_t timeout_ms)
509 {
510 using namespace std::chrono;
511 auto start_time = steady_clock::now();
512 while (requested_msg_id_ != REQUEST_PENDING) {
513 can_intf_.pump_events(); // pump event loop while waiting
514 auto elapsed_time = duration_cast<milliseconds>(steady_clock::now() - start_time);
515 if (elapsed_time.count() > timeout_ms) return false;
516 }
517 return true;
518 }

PID制御を行うクラス

例(mbed)

// 使用例
roblab::common_lib::Pid pid{0.02f}; // 制御周期0.02s
pid.SetGain(0.1, 0.01, 0.01);
pid.SetMaxOutput(10);
pid.SetTarget(360); // 目標値360°
while (true)
{
motor.Drive(pid.Control(encoder.GetDeg()));
ThisThread::sleep_for(20ms);
}

54template <typename T>
55class DataSerializer
56{
57public:
64 static void Serialize(const T data, etl::icircular_buffer<uint8_t> & output)
65 {
66 static_assert(
67 SerializeTypeInfo<T>::IsSerializable() && SerializeTypeInfo<T>::IsSingle(),
68 "T is not supported type for conversion.");
69
70 const uint8_t * bytes = reinterpret_cast<const uint8_t *>(&data);
71
72 // if constexpr
73 if (sizeof(T) == 1) // 1バイトの場合
74 {
75 output.push(bytes[0]); // 1バイトを追加してすぐ返す
76 return;
77 }
78
79 if (IsLittleEndian()) // ビッグエンディアンに揃える
80 {
81 for (int i = sizeof(T) - 1; i >= 0; --i) {
82 output.push(bytes[i]);
83 }
84 } else {
85 for (unsigned int i = 0; i < sizeof(T); ++i) {
86 output.push(bytes[i]);
87 }
88 }
89 }
90
99 static bool Deserialize(etl::icircular_buffer<uint8_t> & target_bytes, T & output)
100 {
101 static_assert(
102 SerializeTypeInfo<T>::IsSerializable() && SerializeTypeInfo<T>::IsSingle(),
103 "T is not supported type for conversion.");
104
105 // バイト列のサイズチェック
106 if (target_bytes.size() < sizeof(T)) {
107 return false;
108 }
109
110 // 1バイト刻みで扱う
111 uint8_t * bytes = reinterpret_cast<uint8_t *>(&output);
112
113 if (sizeof(T) == 1) // 1バイトデータの場合
114 {
115 bytes[0] = target_bytes.front();
116 target_bytes.pop();
117 } else // 複数バイトデータの場合
118 {
119 // エンディアン変換
120 if (IsLittleEndian()) {
121 for (int i = sizeof(T) - 1; i >= 0; --i) {
122 bytes[i] = target_bytes.front();
123 target_bytes.pop();
124 }
125 } else {
126 for (unsigned int i = 0; i < sizeof(T); ++i) {
127 bytes[i] = target_bytes.front();
128 target_bytes.pop();
129 }
130 }
131 }
132
133 return true;
134 }
135};

55 static ByteRingBuffer Serialize(T data)
56 {
57 static_assert(SerializeTypeInfo<T>::IsSerializable() && SerializeTypeInfo<T>::IsSingle(), "T is not supported type for conversion.");
58
59 if (sizeof(T) == 1) // 1バイトの場合
60 {
61 return ByteRingBuffer{*reinterpret_cast<uint8_t *>(&data)}; // 1要素のバイト列を作ってすぐ返す
62 }
63
64 std::vector<uint8_t> vec{};
65 vec.reserve(sizeof(T));
66 uint8_t *bytes = reinterpret_cast<uint8_t *>(&data);
67
68 if (IsLittleEndian()) // ビッグエンディアンに揃える
69 {
70 for (int i = sizeof(T) - 1; i >= 0; --i)
71 {
72 vec.push_back(bytes[i]);
73 }
74 }
75 else
76 {
77 for (int i = 0; i < sizeof(T); ++i)
78 {
79 vec.push_back(bytes[i]);
80 }
81 }
82
83 return ByteRingBuffer{std::move(vec)};
84 }

92 static T Deserialize(ByteRingBuffer &bytes)
93 {
94 static_assert(SerializeTypeInfo<T>::IsSerializable() && SerializeTypeInfo<T>::IsSingle(), "T is not supported type for conversion.");
95
96 if (bytes.Size() < sizeof(T))
97 {
98 // printf("invalid bytes\n");
99 return T{};
100 }
101 assert(!bytes.Empty());
102
103 T data;
104
105 if (sizeof(T) == 1) // 1バイトデータの場合
106 {
107 std::memcpy(&data, &bytes[0], 1);
108 }
109 else // 複数バイトデータの場合
110 {
111 // ビッグエンディアンに揃える
112 std::vector<uint8_t> vec{};
113 vec.reserve(sizeof(T));
114 if (IsLittleEndian())
115 {
116 for (int i = sizeof(T) - 1; i >= 0; --i)
117 {
118 vec.push_back(bytes[i]);
119 }
120 }
121 else
122 {
123 for (int i = 0; i < sizeof(T); ++i)
124 {
125 vec.push_back(bytes.GetFront());
126 }
127 }
128 std::memcpy(&data, vec.data(), sizeof(T));
129 }
130 bytes.PopFront(sizeof(T));
131
132 return data;
133 }

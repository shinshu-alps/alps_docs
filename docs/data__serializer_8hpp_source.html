61 static type::ByteRingBuffer Serialize(T data)
62 {
63 static_assert(SerializeTypeInfo<T>::IsSerializable() && SerializeTypeInfo<T>::IsSingle(), "T is not supported type for conversion.");
64
65 if (sizeof(T) == 1) // 1バイトの場合
66 {
67 return type::ByteRingBuffer{*reinterpret_cast<uint8_t *>(&data)}; // 1要素のバイト列を作ってすぐ返す
68 }
69
70 std::vector<uint8_t> vec{};
71 vec.reserve(sizeof(T));
72 uint8_t *bytes = reinterpret_cast<uint8_t *>(&data);
73
74 if (IsLittleEndian()) // ビッグエンディアンに揃える
75 {
76 for (int i = sizeof(T) - 1; i >= 0; --i)
77 {
78 vec.push_back(bytes[i]);
79 }
80 }
81 else
82 {
83 for (unsigned int i = 0; i < sizeof(T); ++i)
84 {
85 vec.push_back(bytes[i]);
86 }
87 }
88
89 return type::ByteRingBuffer{std::move(vec)};
90 }

98 static T Deserialize(type::ByteRingBuffer &bytes)
99 {
100 static_assert(SerializeTypeInfo<T>::IsSerializable() && SerializeTypeInfo<T>::IsSingle(), "T is not supported type for conversion.");
101
102 if (bytes.Size() < sizeof(T))
103 {
104 // printf("invalid bytes\n");
105 return T{};
106 }
107 assert(!bytes.Empty());
108
109 T data;
110
111 if (sizeof(T) == 1) // 1バイトデータの場合
112 {
113 std::memcpy(&data, &bytes[0], 1);
114 }
115 else // 複数バイトデータの場合
116 {
117 // ビッグエンディアンに揃える
118 std::vector<uint8_t> vec{};
119 vec.reserve(sizeof(T));
120 if (IsLittleEndian())
121 {
122 for (int i = sizeof(T) - 1; i >= 0; --i)
123 {
124 vec.push_back(bytes[i]);
125 }
126 }
127 else
128 {
129 for (unsigned int i = 0; i < sizeof(T); ++i)
130 {
131 vec.push_back(bytes.GetFront());
132 }
133 }
134 std::memcpy(&data, vec.data(), sizeof(T));
135 }
136 bytes.PopFront(sizeof(T));
137
138 return data;
139 }

155 public:
162 static type::ByteRingBuffer Serialize(const std::vector<U> &vector_data)
163 {
164 static_assert(SerializeTypeInfo<U>::IsSerializable(), "U in std::vector<U> is not supported conversion.");
165
166 // 前もって領域を確保
167 type::ByteRingBuffer bytes{};
168 bytes.Reserve(sizeof(vector_size_t) + sizeof(U) * vector_data.size());
169
170 // 要素数
171 vector_size_t vec_size = vector_data.size();
172 bytes.PushBack(DataSerializer<vector_size_t>::Serialize(vec_size));
173
174 // 要素
175 for (const auto &data : vector_data)
176 {
177 bytes.PushBack(DataSerializer<U>::Serialize(data));
178 }
179 return bytes;
180 }

196

200 data.push_back(DataSerializer<U>::Deserialize(bytes));

226 static type::ByteRingBuffer Serialize(const std::string &string_data)
227 {
228 // 前もって領域を確保
229 type::ByteRingBuffer bytes{};
230 bytes.Reserve(sizeof(string_size_t) + string_data.size());
231
232 // 文字数
233 string_size_t str_size = string_data.size();
234 bytes.PushBack(DataSerializer<string_size_t>::Serialize(str_size));
235
236 // 文字列内容
237 std::vector<uint8_t> str_bytes(string_data.size());
238 std::memcpy(str_bytes.data(), string_data.c_str(), string_data.size());
239 bytes.PushBack(type::ByteRingBuffer{std::move(str_bytes)});
240
241 return bytes;
242 }

250 static std::string Deserialize(type::ByteRingBuffer &bytes)
251 {
252 // 文字数取得
253 string_size_t string_size = DataSerializer<string_size_t>::Deserialize(bytes);
254
255 // 文字数分のバイトが存在しているか
256 if (bytes.Size() < string_size)
257 {
258 // printf("lack of bytes for string num.\n");
259 return std::string{};
260 }
261
262 // 文字列ポインタからのコンストラクタを利用
263 std::string ret_str(reinterpret_cast<char *>(bytes.ToVector().data()), string_size);
264 bytes.PopFront(string_size);
265 return ret_str;
266 }

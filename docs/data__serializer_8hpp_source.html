53 template <typename T>
54 class DataSerializer
55 {
56 public:
63 static void Serialize(const T data, etl::icircular_buffer<uint8_t> &output)
64 {
65 static_assert(SerializeTypeInfo<T>::IsSerializable() && SerializeTypeInfo<T>::IsSingle(), "T is not supported type for conversion.");
66
67 const uint8_t *bytes = reinterpret_cast<const uint8_t *>(&data);
68
69 // if constexpr
70 if (sizeof(T) == 1) // 1バイトの場合
71 {
72 output.push(bytes[0]); // 1バイトを追加してすぐ返す
73 return;
74 }
75
76 if (IsLittleEndian()) // ビッグエンディアンに揃える
77 {
78 for (int i = sizeof(T) - 1; i >= 0; --i)
79 {
80 output.push(bytes[i]);
81 }
82 }
83 else
84 {
85 for (unsigned int i = 0; i < sizeof(T); ++i)
86 {
87 output.push(bytes[i]);
88 }
89 }

100 static bool Deserialize(etl::icircular_buffer<uint8_t> &target_bytes, T &output)
101 {
102 static_assert(SerializeTypeInfo<T>::IsSerializable() && SerializeTypeInfo<T>::IsSingle(), "T is not supported type for conversion.");
103
104 // バイト列のサイズチェック
105 if (target_bytes.size() < sizeof(T))
106 {
107 return false;
108 }
109
110 // 1バイト刻みで扱う
111 uint8_t *bytes = reinterpret_cast<uint8_t *>(&output);
112
113 if (sizeof(T) == 1) // 1バイトデータの場合
114 {
115 bytes[0] = target_bytes.front();
116 target_bytes.pop();
117 }
118 else // 複数バイトデータの場合
119 {
120 // エンディアン変換
121 if (IsLittleEndian())
122 {
123 for (int i = sizeof(T) - 1; i >= 0; --i)
124 {
125 bytes[i] = target_bytes.front();
126 target_bytes.pop();
127 }
128 }
129 else
130 {
131 for (unsigned int i = 0; i < sizeof(T); ++i)
132 {
133 bytes[i] = target_bytes.front();
134 target_bytes.pop();
135 }
136 }
137 }
138
139 return true;
140 }

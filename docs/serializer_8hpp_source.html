94  {

107  static bool ToBytes(const T & data, etl::icircular_buffer<uint8_t> & output)
108  {
109  // シリアライズしたデータを格納する空き容量があるかチェック
110  if (output.available() < RequiredByteSize()) {
111  return false; // 失敗
112  }
113 
114  const uint8_t * bytes = reinterpret_cast<const uint8_t *>(&data);
115 
116  // if constexpr
117  if (sizeof(T) == 1) // 1バイトの場合
118  {
119  output.push(bytes[0]); // 1バイトを追加してすぐ返す
120  return true; // 成功
121  }
122 
123  if (detail::IsLittleEndian()) // ビッグエンディアンに揃える
124  {
125  for (int i = sizeof(T) - 1; i >= 0; --i) {
126  output.push(bytes[i]);
127  }
128  } else {
129  for (unsigned int i = 0; i < sizeof(T); ++i) {
130  output.push(bytes[i]);
131  }
132  }
133  return true; // 成功
134  }

144  static bool ToOriginal(etl::icircular_buffer<uint8_t> & target_bytes, T & output)
145  {
146  // バイト列のサイズチェック
147  if (target_bytes.size() < sizeof(T)) {
148  return false; // 失敗
149  }
150 
151  // 1バイト刻みで扱う
152  uint8_t * bytes = reinterpret_cast<uint8_t *>(&output);
153 
154  if (sizeof(T) == 1) // 1バイトデータの場合
155  {
156  bytes[0] = target_bytes.front();
157  target_bytes.pop();
158  } else // 複数バイトデータの場合
159  {
160  // エンディアン変換
161  if (detail::IsLittleEndian()) {
162  for (int i = sizeof(T) - 1; i >= 0; --i) {
163  bytes[i] = target_bytes.front();
164  target_bytes.pop();
165  }
166  } else {
167  for (unsigned int i = 0; i < sizeof(T); ++i) {
168  bytes[i] = target_bytes.front();
169  target_bytes.pop();
170  }
171  }
172  }
173 
174  return true; // 成功
175  }

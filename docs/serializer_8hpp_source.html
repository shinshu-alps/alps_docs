106  static bool ToBytes(const T & data, etl::icircular_buffer<uint8_t> & output)
107  {
108  // シリアライズしたデータを格納する空き容量があるかチェック
109  if (output.available() < RequiredByteSize()) {
110  return false; // 失敗
111  }
112 
113  const uint8_t * bytes = reinterpret_cast<const uint8_t *>(&data);
114 
115  // if constexpr
116  if (sizeof(T) == 1) // 1バイトの場合
117  {
118  output.push(bytes[0]); // 1バイトを追加してすぐ返す
119  return true; // 成功
120  }
121 
122  if (detail::IsLittleEndian()) // ビッグエンディアンに揃える
123  {
124  for (int i = sizeof(T) - 1; i >= 0; --i) {
125  output.push(bytes[i]);
126  }
127  } else {
128  for (unsigned int i = 0; i < sizeof(T); ++i) {
129  output.push(bytes[i]);
130  }
131  }
132  return true; // 成功
133  }

143  static bool ToOriginal(etl::icircular_buffer<uint8_t> & target_bytes, T & output)
144  {
145  // バイト列のサイズチェック
146  if (target_bytes.size() < sizeof(T)) {
147  return false; // 失敗
148  }
149 
150  // 1バイト刻みで扱う
151  uint8_t * bytes = reinterpret_cast<uint8_t *>(&output);
152 
153  if (sizeof(T) == 1) // 1バイトデータの場合
154  {
155  bytes[0] = target_bytes.front();
156  target_bytes.pop();

159  // エンディアン変換

162  bytes[i] = target_bytes.front();
163  target_bytes.pop();
164  }
165  } else {
166  for (unsigned int i = 0; i < sizeof(T); ++i) {
167  bytes[i] = target_bytes.front();
168  target_bytes.pop();
169  }
170  }
171  }
172 
173  return true; // 成功

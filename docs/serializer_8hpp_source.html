105  static bool Serialize(const T & data, etl::icircular_buffer<uint8_t> & output)
106  {
107  // シリアライズしたデータを格納する空き容量があるかチェック
108  if (output.available() < RequiredByteSize()) {
109  return false; // 失敗
110  }
111 
112  const uint8_t * bytes = reinterpret_cast<const uint8_t *>(&data);
113 
114  // if constexpr
115  if (sizeof(T) == 1) // 1バイトの場合
116  {
117  output.push(bytes[0]); // 1バイトを追加してすぐ返す
118  return true; // 成功
119  }
120 
121  if (detail::IsLittleEndian()) // ビッグエンディアンに揃える
122  {
123  for (int i = sizeof(T) - 1; i >= 0; --i) {
124  output.push(bytes[i]);
125  }
126  } else {
127  for (unsigned int i = 0; i < sizeof(T); ++i) {
128  output.push(bytes[i]);
129  }
130  }
131  return true; // 成功
132  }

142  static bool Deserialize(etl::icircular_buffer<uint8_t> & target_bytes, T & output)
143  {
144  // バイト列のサイズチェック
145  if (target_bytes.size() < sizeof(T)) {
146  return false; // 失敗
147  }
148 
149  // 1バイト刻みで扱う
150  uint8_t * bytes = reinterpret_cast<uint8_t *>(&output);
151 
152  if (sizeof(T) == 1) // 1バイトデータの場合
153  {
154  bytes[0] = target_bytes.front();
155  target_bytes.pop();
156  } else // 複数バイトデータの場合
157  {
158  // エンディアン変換

161  bytes[i] = target_bytes.front();
162  target_bytes.pop();
163  }
164  } else {
165  for (unsigned int i = 0; i < sizeof(T); ++i) {
166  bytes[i] = target_bytes.front();
167  target_bytes.pop();
168  }
169  }
170  }
171 
172  return true; // 成功

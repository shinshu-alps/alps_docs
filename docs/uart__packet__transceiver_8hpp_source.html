79 template <class UartIo>
80 bool WritePacket(IUartIo<UartIo> & uart_io)
81 {
82 // 新たに値がセットされていない場合は何もしない
83 if (!has_new_val_) {
84 return true;
85 }
86
87 // IDとデータからパケットを作成
88 etl::circular_buffer<uint8_t, UartPacketByteSize<T>()> packet;
89 type::Serializer<UartPacketId>::ToBytes(id, packet);
90 type::Serializer<T>::ToBytes(data_, packet);
91 Cobs::Encode(packet);
92
93 // 送信
94 ssize_t send_size = uart_io.CallWrite(packet);
95
96 if (send_size != static_cast<ssize_t>(packet.size())) {
97 return false;
98 }
99
100 has_new_val_ = false;
101 return true;
102 }
103
104public:
106 const UartPacketId id;
107
108private:
110 T data_{};
112 bool has_new_val_{false};
113};

226 bool WriteUpdate()
227 {
228 for (auto & func : write_packet_funcs_) {
229 // 登録された書き込み処理を呼び出す
230 auto result = func(uart_io_);
231 if (!result) {
232 return false;
233 }
234 }
235 return true;

242 void ReadUpdate()
243 {
244 auto result = uart_io_.CallRead(read_buf_);
245
246 if (result <= 0) // 何も読み取らなかった，またはエラー時は終了
247 {
248 return;
249 }
250
251 // デシリアライズ用の一時バッファ
252 etl::circular_buffer<uint8_t, kMaxDataByteSize> tmp_read_data_buf_{};
253
254 while (true) {
255 tmp_read_data_buf_.clear();
256
257 // COBSの終端がある場合はパケットとして処理する
258 if (!Cobs::TryDecode(read_buf_, tmp_read_data_buf_)) break;
259
260 UartPacketId id;
261 if (type::Serializer<UartPacketId>::ToOriginal(tmp_read_data_buf_, id)) {
262 if (IsUsedReceiveId(id)) {
263 receive_data_callbacks_[id](tmp_read_data_buf_);
264 }
265 }
266 }
267 }

282 {
283 return receive_data_callbacks_.count(id) != 0;
284 }

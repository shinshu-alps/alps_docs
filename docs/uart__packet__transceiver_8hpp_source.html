79 template <class UartIo>
80 bool WritePacket(IUartIo<UartIo> & uart_io)
81 {
82 // 新たに値がセットされていない場合は何もしない
83 if (!has_new_val_) {
84 return true;
85 }
86
87 // IDとデータからパケットを作成
88 etl::circular_buffer<uint8_t, UartPacketByteSize<T>()> packet;
89 DataSerializer<UartPacketId>::Serialize(id, packet);
90 DataSerializer<T>::Serialize(data_, packet);
91 Cobs::Encode(packet);
92
93 // 送信
94 ssize_t send_size = uart_io.CallWrite(packet);
95
96 if (send_size != static_cast<ssize_t>(packet.size())) {
97 return false;
98 }
99
100 has_new_val_ = false;
101 return true;
102 }
103
104public:
106 const UartPacketId id;
107
108private:
110 T data_{};
112 bool has_new_val_{false};
113};

200 template <typename T>
201 bool RegisterUartReceiver(UartReceiver<T> & receiver)
202 {
203 if (IsUsedReceiveId(receiver.id_)) // 予約済みIDの場合
204 {
205 return false;
206 }
207
208 // コールバック処理を登録
209 receive_data_callbacks_[receiver.id_] = etl::delegate<void(etl::icircular_buffer<uint8_t> &)>::
210 create<UartReceiver<T>, &UartReceiver<T>::OnReceiveData>(receiver);
211
212 return true;
213 }
214
221 bool WriteUpdate()
222 {
223 for (auto & func : write_packet_funcs_) {
224 // 登録された書き込み処理を呼び出す
225 auto result = func(uart_io_);
226 if (!result) {
227 return false;
228 }
229 }
230 return true;
231 }
232
237 void ReadUpdate()
238 {
239 auto result = uart_io_.CallRead(read_buf_);
240
241 if (result <= 0) // 何も読み取らなかった，またはエラー時は終了
242 {
243 return;
244 }
245
246 // デシリアライズ用の一時バッファ
247 etl::circular_buffer<uint8_t, kMaxDataByteSize> tmp_read_data_buf_{};
248
249 while (true) {
250 tmp_read_data_buf_.clear();
251
252 // COBSの終端がある場合はパケットとして処理する
253 if (!Cobs::TryDecode(read_buf_, tmp_read_data_buf_)) break;
254
255 UartPacketId id;
256 if (DataSerializer<UartPacketId>::Deserialize(tmp_read_data_buf_, id)) {
257 if (IsUsedReceiveId(id)) {
258 receive_data_callbacks_[id](tmp_read_data_buf_);
259 }
260 }
261 }
262 }
263
264private:
268
276 bool IsUsedReceiveId(UartPacketId id)
277 {
278 return receive_data_callbacks_.count(id) != 0;
279 }
280
281private:
283 IUartIo<UartIo> & uart_io_;
284
286 etl::circular_buffer<uint8_t, kReadBufferSize> read_buf_;
287
289 etl::vector<etl::delegate<bool(IUartIo<UartIo> &)>, kMaxSenderNum> write_packet_funcs_;
290
292 etl::unordered_map<
293 UartPacketId, etl::delegate<void(etl::icircular_buffer<uint8_t> &)>, kMaxReceiveNum>
294 receive_data_callbacks_;
295};

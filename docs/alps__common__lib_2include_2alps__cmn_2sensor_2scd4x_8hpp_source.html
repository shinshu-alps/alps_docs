836 static uint8_t CalcCrc8(uint8_t msb, uint8_t lsb)
837 {
838 uint8_t crc = 0xFFu;
839 crc ^= msb;
840 for (int i = 0; i < 8; ++i) {
841 crc = (crc & 0x80u) ? static_cast<uint8_t>((crc << 1) ^ 0x31u)
842 : static_cast<uint8_t>(crc << 1);
843 }
844 crc ^= lsb;
845 for (int i = 0; i < 8; ++i) {
846 crc = (crc & 0x80u) ? static_cast<uint8_t>((crc << 1) ^ 0x31u)
847 : static_cast<uint8_t>(crc << 1);
848 }
849 return crc;
850 }

878 template <size_t N>
879 bool WriteCommandWithArgs(Scd4xCmdId cmd_id, const std::array<uint16_t, N> & args) {
880 const uint16_t cmd = static_cast<uint16_t>(cmd_id);
881 constexpr size_t kBufSize = 2 + 3 * N; // cmd(2) + (MSB, LSB, CRC) * N
882 uint8_t write_buf[kBufSize];
883
884 write_buf[0] = static_cast<uint8_t>((cmd >> 8) & 0xFFu);
885 write_buf[1] = static_cast<uint8_t>(cmd & 0xFFu);
886
887 size_t idx = 2;
888 for (size_t i = 0; i < N; ++i) {
889 const uint8_t msb = static_cast<uint8_t>((args[i] >> 8) & 0xFFu);
890 const uint8_t lsb = static_cast<uint8_t>(args[i] & 0xFFu);
891 write_buf[idx++] = msb;
892 write_buf[idx++] = lsb;
893 write_buf[idx++] = CalcCrc8(msb, lsb);
894 }
895
896 return i2c_io_.CallWrite(i2c_address_, write_buf);
897 }

909 template <size_t N>
910 bool ReadCommandData(
911 Scd4xCmdId cmd_id,
912 std::array<uint16_t, N> & read_data,
913 uint32_t delay_ms)
914 {
915 const uint16_t cmd = static_cast<uint16_t>(cmd_id);
916 uint8_t write_buf[2] = {
917 static_cast<uint8_t>((cmd >> 8) & 0xFFu),
918 static_cast<uint8_t>(cmd & 0xFFu)
919 };
920
921 if (!i2c_io_.CallWrite(i2c_address_, write_buf)) {
922 return false;
923 }
924
925 if (delay_ms > 0) {
926 alps::cmn::time::SleepFor(std::chrono::milliseconds(delay_ms));
927 }
928
929 constexpr size_t kRxSize = 3 * N; // (MSB, LSB, CRC) * N
930 uint8_t read_buf[kRxSize];
931
932 if (!i2c_io_.CallRead(i2c_address_, etl::span<uint8_t>(read_buf, kRxSize))) {
933 return false;
934 }
935
936 for (size_t i = 0; i < N; ++i) {
937 const size_t base = i * 3;
938 const uint8_t msb = read_buf[base];
939 const uint8_t lsb = read_buf[base + 1];
940 const uint8_t crc = read_buf[base + 2];
941 const uint8_t calc = CalcCrc8(msb, lsb);
942
943 if (crc != calc) {
944 return false; // CRCエラー
945 }
946
947 read_data[i] = static_cast<uint16_t>((msb << 8) | lsb);
948 }
949
950 return true;
951 }

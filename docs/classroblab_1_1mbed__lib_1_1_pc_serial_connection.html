詳解

パソコンとシリアル接続し，ID付きパケット通信を行う

例

float g_output;
void OnReceiveMotorOutput(float &&output)
{
g_output = output;
}
int main(void)
{
roblab::mbed_lib::PcSerialConnection pc("mbed0", 115200);
pc.RegisterReceiveCallback<float>(0, OnReceiveMotorOutput);
auto test_sender = pc.CreateSender<float>(0);
while (true)
{
test_sender.EnqueueSendData(1.2345);
pc.Update();
ThisThread::sleep_for(20ms);
}
}

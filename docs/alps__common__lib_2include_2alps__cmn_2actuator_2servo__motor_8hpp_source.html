133 void SetAngle(uint16_t target_angle_deg)
134 {
135 if (target_angle_deg > max_angle_deg_) {
136 target_angle_deg = max_angle_deg_;
137 }
138 std::chrono::microseconds pulse_width =
139 min_pulse_width_ + std::chrono::duration_cast<std::chrono::microseconds>(
140 (max_pulse_width_ - min_pulse_width_) *
141 (static_cast<float>(target_angle_deg) / max_angle_deg_));
142 pwm_out_.CallSetPulseWidth(pulse_width);
143 }
144
145private:

149 const uint16_t max_angle_deg_;

153 const std::chrono::microseconds max_pulse_width_;

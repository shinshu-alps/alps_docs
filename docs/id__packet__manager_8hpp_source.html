124 type::ByteRingBuffer MakePacket(PacketId id, type::ByteRingBuffer raw_data_bytes)
125 {
126 auto bytes = util::DataSerializer<PacketId>::Serialize(id); // idをヘッダーとしてつける

133 void ParsePacket()
134 {
135 for (auto &packet : Cobs::ExtractCobsPackets(read_buf_))
136 {
137 if (packet.Empty())
138 continue;
139
140 auto data_bytes = Cobs::Decode(packet);
141 PacketId id = util::DataSerializer<PacketId>::Deserialize(data_bytes); // ヘッダーであるidだけ先に取得
142
143 // printf("received id: %d\n", id);
144
145 if (!IsUnusedReceiveId(id)) // IDが登録されている時
146 {
147 // 対応するコールバック関数を呼び出す
148 receive_data_callbacks_[id](std::move(data_bytes));
149 }
150 else // 登録されていない時

152 // printf("received unknown data. id: %d\n", id); // デバッグ用

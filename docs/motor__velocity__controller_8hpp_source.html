134  void Control(float ff_output = 0.0f)
135  {
136  // calc_info_の値は全て更新される
137 
138  // 現在の角速度計算
139  calc_info_.now_velocity = now_velocity_calculator_.Calculate(rotary_sensor_.CallGetAngle());
140  calc_info_.now_velocity_before_lpf = now_velocity_calculator_.GetLastRawOutput();
141  float velocity_error = calc_info_.target_velocity - calc_info_.now_velocity;
142 
143  // 目標角加速度計算
144  calc_info_.target_acceleration =
145  target_acceleration_calculator_.Calculate(calc_info_.target_velocity);
146  calc_info_.target_acceleration_before_lpf = target_acceleration_calculator_.GetLastRawOutput();
147 
148  // フィードフォワード項計算
149  calc_info_.ff_velocity_term = param_.kff_velocity * calc_info_.target_velocity;
150  calc_info_.ff_acceleration_term = param_.kff_acceleration * calc_info_.target_acceleration;
151  float ff_output_term =
152  calc_info_.ff_velocity_term + calc_info_.ff_acceleration_term + ff_output;
153 
154  // PID計算
155  float output = vel_pid_.Calculate(velocity_error, ff_output_term);
156  calc_info_.pid_calc_info = vel_pid_.GetCalcInfo();
157 
158  // モーターを駆動
159  motor_.CallDrive(output);
160  }

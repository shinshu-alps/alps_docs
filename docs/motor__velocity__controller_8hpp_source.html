109class MotorVelocityController

111public:
120 MotorVelocityController(

123 const MotorVelocityControllerParam & param,

125 : motor_{motor},
126 rotary_sensor_{rotary_sensor},

132 }
133
139 void Control(float ff_output = 0.0f)
140 {
141 // calc_info_の値は全て更新される
142
143 // 現在の角速度計算
144 calc_info_.now_velocity = now_velocity_calculator_.Calculate(rotary_sensor_.CallGetAngle());
145 calc_info_.now_velocity_before_lpf = now_velocity_calculator_.GetLastRawOutput();
146 float velocity_error = calc_info_.target_velocity - calc_info_.now_velocity;
147
148 // 目標角加速度計算
149 calc_info_.target_acceleration =
150 target_acceleration_calculator_.Calculate(calc_info_.target_velocity);
151 calc_info_.target_acceleration_before_lpf = target_acceleration_calculator_.GetLastRawOutput();
152
153 // フィードフォワード項計算
154 calc_info_.ff_velocity_term = param_.kff_velocity * calc_info_.target_velocity;
155 calc_info_.ff_acceleration_term = param_.kff_acceleration * calc_info_.target_acceleration;
156 float ff_output_term =
157 calc_info_.ff_velocity_term + calc_info_.ff_acceleration_term + ff_output;
158
159 // PID計算
160 float output = vel_pid_.Calculate(velocity_error, ff_output_term);
161 calc_info_.pid_calc_info = vel_pid_.GetCalcInfo();
162
163 // モーターを駆動
164 motor_.CallDrive(output);
165 }

171 void Reset()

173 now_velocity_calculator_.Reset();

175 vel_pid_.Reset();
176 }

102 void Control(float ff_output = 0.0f)
103 {
104 // calc_info_の値は全て更新される
105
106 // 現在の角速度計算
107 calc_info_.now_velocity = now_velocity_calculator_.Calculate(rotary_sensor_.CallGetAngle());
108 calc_info_.now_velocity_before_lpf = now_velocity_calculator_.GetLastRawOutput();
109 float velocity_error = calc_info_.target_velocity - calc_info_.now_velocity;
110
111 // 目標角加速度計算
112 calc_info_.target_acceleration =
113 target_acceleration_calculator_.Calculate(calc_info_.target_velocity);
114 calc_info_.target_acceleration_before_lpf = target_acceleration_calculator_.GetLastRawOutput();
115
116 // フィードフォワード項計算
117 calc_info_.ff_velocity_term = param_.kff_velocity * calc_info_.target_velocity;
118 calc_info_.ff_acceleration_term = param_.kff_acceleration * calc_info_.target_acceleration;
119 float ff_output_term =
120 calc_info_.ff_velocity_term + calc_info_.ff_acceleration_term + ff_output;
121
122 // PID計算
123 float output = vel_pid_.Calculate(velocity_error, ff_output_term);
124 calc_info_.pid_calc_info = vel_pid_.GetCalcInfo();
125
126 // モーターを駆動
127 motor_.CallDrive(output);
128 }

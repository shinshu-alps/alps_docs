162 void IncrementPulses(bool a_state)
163 {
164 int8_t b_state = read_B_->Read();
165 curr_state_ = (static_cast<int8_t>(a_state) << 1) | (b_state);
166
167 if ((prev_state_ == 0x3 && curr_state_ == 0x0) ||
168 (prev_state_ == 0x0 && curr_state_ == 0x3))
169 {
170 pulses_++;
171 }
172 else if ((prev_state_ == 0x2 && curr_state_ == 0x1) ||
173 (prev_state_ == 0x1 && curr_state_ == 0x2))
174 {
175 pulses_--;
176 }
177
178 prev_state_ = curr_state_;
179 }

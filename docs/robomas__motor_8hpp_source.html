151  void RegisterMotor(
152  Id id,
153  etl::delegate<int16_t()> motor_output_getter,
154  etl::delegate<void(const detail::SensorCanData & sensor_data)> receive_sensor_data_callback)
155  {
156  // モーターIDの重複チェック
157  if (motor_output_getter_.count(id) != 0) {
158  ALPS_LOG_WARN(
159  "Motor ID: %d is duplicated. This motor is not registered.", static_cast<uint8_t>(id));
160  return;
161  }
162 
163  // モーターを登録
164  motor_output_getter_.insert({id, motor_output_getter});
165 
166  // センサー値を受け取るためのコールバック関数を登録
167  sensor_subscribers_[static_cast<uint8_t>(id) - 1].SetReceiveDataCallback(
168  receive_sensor_data_callback);
169  }
170 
175  void PublishMotorOutput()
176  {
177  std::array<detail::OutputCanData, 2> output_data{};
178 
179  // それぞれのモーターの出力値を配列に格納
180  for (auto & motor : motor_output_getter_) {
181  // motor.second->GetOutput(output_data);
182  Id motor_id = motor.first;
183  int16_t target_current = motor.second();
184 
185  // モーターIDが1~4の場合と5~8の場合で書き込み先のCANフレーム(ID）が異なる
186  if (static_cast<int>(motor_id) <= 4) {
187  output_data[0][((static_cast<int>(motor_id)) - 1) * 2] = target_current >> 8;
188  output_data[0][((static_cast<int>(motor_id)) - 1) * 2 + 1] = target_current & 0xff;
189  } else {
190  output_data[1][((static_cast<int>(motor_id) - 4) - 1) * 2] = target_current >> 8;
191  output_data[1][((static_cast<int>(motor_id) - 4) - 1) * 2 + 1] = target_current & 0xff;
192  }
193  }
194 
195  // CAN送信
196  motor_output_publishers_[0].Publish(output_data[0]);
197  motor_output_publishers_[1].Publish(output_data[1]);
198  }
199 
200 private:
202  etl::unordered_map<Id, etl::delegate<int16_t()>, detail::kMaxMotorNum> motor_output_getter_;
204  std::array<communication::CanPublisher<detail::OutputCanData>, 2> motor_output_publishers_;
206  std::array<communication::CanSubscriber<detail::SensorCanData>, detail::kMaxMotorNum>
207  sensor_subscribers_;
208 };

259  void Drive(float output_ratio)
260  {
261  output_ratio = util::Clamp(output_ratio, -1.0f, 1.0f);
262  target_current_ = sign_ * max_abs_output_ * output_ratio;

280  void OnReceiveSensorFrame(const detail::SensorCanData & sensor_data)
281  {
282  // センサー値の更新
283  last_abs_angle_ = now_abs_angle_;
284  now_abs_angle_ = sensor_data[0] << 8 | sensor_data[1];
285  speed_rpm_ = sensor_data[2] << 8 | sensor_data[3];
286  real_current_ = (sensor_data[4] << 8 | sensor_data[5]);
287 
288  // オフセット角度の設定
289  if (reset_angle_flag_) {
290  last_abs_angle_ = now_abs_angle_;
291  offset_angle_ = now_abs_angle_;
292  reset_angle_flag_ = false;
293  }
294 
295  // 回転数のカウント
296  if (now_abs_angle_ - last_abs_angle_ > 4096)
297  --round_count_;
298  else if (now_abs_angle_ - last_abs_angle_ < -4096)
299  ++round_count_;
300 
301  // 総角度の計算
302  total_angle_ = round_count_ * 8192 + now_abs_angle_ - offset_angle_;
303 
304  // タイムアウトカウントのリセット
305  timeout_checker_.ResetCount();
306  }
307 
310  double GetAngle()
311  {
312  return sign_ * total_angle_ * 2 * alps::cmn::util::kPi / 8192.0 / gear_ratio_;
313  }
314 
317  float GetVelocity()
318  {
319  return sign_ * speed_rpm_ * 2 * alps::cmn::util::kPi / 60.0f / gear_ratio_;
320  }
321 
324  int16_t GetRealCurrent()
325  {

327  }

330  void ResetAngle()

333  round_count_ = 0;
334  reset_angle_flag_ = true;
335  }

343  }

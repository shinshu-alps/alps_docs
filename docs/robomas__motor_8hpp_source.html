159 void RegisterMotor(
160 Id id,
161 etl::delegate<int16_t()> motor_output_getter,
162 etl::delegate<void(const detail::SensorCanData & sensor_data)> receive_sensor_data_callback)
163 {
164 // モーターIDの重複チェック
165 if (motor_output_getter_.count(id) != 0) {
166 ALPS_LOG_WARN(
167 "Motor ID: %d is duplicated. This motor is not registered.", static_cast<uint8_t>(id));
168 return;
169 }
170
171 // モーターを登録
172 motor_output_getter_.insert({id, motor_output_getter});
173
174 // センサー値を受け取るためのコールバック関数を登録
175 sensor_subscribers_[static_cast<uint8_t>(id) - 1].SetReceiveDataCallback(
176 receive_sensor_data_callback);
177 }

183 void PublishMotorOutput()
184 {
185 std::array<detail::OutputCanData, 2> output_data{};
186
187 // それぞれのモーターの出力値を配列に格納
188 for (auto & motor : motor_output_getter_) {
189 // motor.second->GetOutput(output_data);
190 Id motor_id = motor.first;
191 int16_t target_current = motor.second();
192
193 // モーターIDが1~4の場合と5~8の場合で書き込み先のCANフレーム(ID）が異なる
194 if (static_cast<int>(motor_id) <= 4) {
195 output_data[0][((static_cast<int>(motor_id)) - 1) * 2] = target_current >> 8;
196 output_data[0][((static_cast<int>(motor_id)) - 1) * 2 + 1] = target_current & 0xff;
197 } else {
198 output_data[1][((static_cast<int>(motor_id) - 4) - 1) * 2] = target_current >> 8;
199 output_data[1][((static_cast<int>(motor_id) - 4) - 1) * 2 + 1] = target_current & 0xff;
200 }
201 }
202
203 // CAN送信
204 motor_output_publishers_[0].Publish(output_data[0]);
205 motor_output_publishers_[1].Publish(output_data[1]);
206 }
207
208private:
210 etl::unordered_map<Id, etl::delegate<int16_t()>, detail::kMaxMotorNum> motor_output_getter_;
212 std::array<communication::CanPublisher<detail::OutputCanData>, 2> motor_output_publishers_;
214 std::array<communication::CanSubscriber<detail::SensorCanData>, detail::kMaxMotorNum>
215 sensor_subscribers_;

298 void OnReceiveSensorFrame(const detail::SensorCanData & sensor_data)
299 {
300 // センサー値の更新
301 last_abs_angle_ = now_abs_angle_;
302 now_abs_angle_ = sensor_data[0] << 8 | sensor_data[1];
303 speed_rpm_ = sensor_data[2] << 8 | sensor_data[3];
304 real_current_ = (sensor_data[4] << 8 | sensor_data[5]);
305
306 // オフセット角度の設定
307 if (reset_angle_flag_) {
308 last_abs_angle_ = now_abs_angle_;
309 offset_angle_ = now_abs_angle_;
310 reset_angle_flag_ = false;
311 }
312
313 // 回転数のカウント
314 if (now_abs_angle_ - last_abs_angle_ > 4096)
315 --round_count_;
316 else if (now_abs_angle_ - last_abs_angle_ < -4096)
317 ++round_count_;
318
319 // 総角度の計算
320 total_angle_ = round_count_ * 8192 + now_abs_angle_ - offset_angle_;
321
322 // タイムアウトカウントのリセット
323 timeout_checker_.ResetCount();
324
325 // コールバックの呼び出し
326 on_receive_sensor_frame_callback_.call_if();
327 }

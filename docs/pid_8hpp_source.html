102  {
103  // 偏差を更新

105  calc_info_.error_diff = low_pass_differentiator_.Calculate(error);
106  calc_info_.error_diff_before_lpf = low_pass_differentiator_.GetLastRawOutput();
107 
108  // フィードフォワード

110 
111  // 比例
112  calc_info_.p_term = param_.kp * error;
113 
114  // 積分
115  // Anti-windup処理．出力値が飽和している場合は積分を行わない．
116  // 参考: https://www.youtube.com/watch?v=NVLXCwc8HzM&ab_channel=MATLAB
117  bool is_enable_integrator = !(is_saturated_ && error * calc_info_.output > 0.0f);
118  if (is_enable_integrator) {
119  calc_info_.i_term = param_.ki * integrator_.Calculate(error);
120  } else {
121  calc_info_.i_term = param_.ki * integrator_.Calculate(0);
122  }
123 
124  // 微分
125  // ローパスフィルタをかけたD出力
126  calc_info_.d_term = param_.kd * calc_info_.error_diff;
127 
128  // PID出力を計算
129  float output_before_clamp =
130  calc_info_.p_term + calc_info_.i_term + calc_info_.d_term + calc_info_.ff_term;
131  calc_info_.output = output_limit_.Clamp(output_before_clamp);
132 
133  // 飽和判定

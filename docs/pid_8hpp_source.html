131  float Calculate(float error, float ff_output = 0.0f)
132  {
133  // 偏差を更新
134  calc_info_.error = error;
135  calc_info_.error_diff = low_pass_differentiator_.Calculate(error);
136  calc_info_.error_diff_before_lpf = low_pass_differentiator_.GetLastRawOutput();
137 
138  // フィードフォワード
139  calc_info_.ff_term = ff_output;
140 
141  // 比例
142  calc_info_.p_term = param_.kp * error;
143 
144  // 積分
145  // Anti-windup処理．出力値が飽和している場合は積分を行わない．
146  // 参考: https://www.youtube.com/watch?v=NVLXCwc8HzM&ab_channel=MATLAB
147  bool is_enable_integrator = !(is_saturated_ && error * calc_info_.output > 0.0f);
148  if (is_enable_integrator) {
149  calc_info_.i_term = param_.ki * integrator_.Calculate(error);
150  } else {
151  calc_info_.i_term = param_.ki * integrator_.Calculate(0);
152  }
153 
154  // 微分
155  // ローパスフィルタをかけたD出力
156  calc_info_.d_term = param_.kd * calc_info_.error_diff;
157 
158  // PID出力を計算
159  float output_before_clamp =
160  calc_info_.p_term + calc_info_.i_term + calc_info_.d_term + calc_info_.ff_term;
161  calc_info_.output = output_limit_.Clamp(output_before_clamp);
162 
163  // 飽和判定
164  is_saturated_ = (output_before_clamp != calc_info_.output);
165 
166  return calc_info_.output;
167  }

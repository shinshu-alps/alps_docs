102 
103  // フィードフォワード項

105 
106  // 比例

108 
109  // 積分
110  // Anti-windup処理．参考: https://www.youtube.com/watch?v=NVLXCwc8HzM&ab_channel=MATLAB
111  bool is_enable_integrator = !(is_saturated_ && error * pid_result_.output > 0.0f);
112  if (is_enable_integrator) {
113  pid_result_.i_term = pid_param_.ki * integrator_.Calculate(error);
114  } else {
115  pid_result_.i_term = pid_param_.ki * integrator_.Calculate(0);
116  }
117 
118  // 微分
119  // ローパスフィルタをかけたD出力
120  pid_result_.d_term = pid_param_.kd * low_pass_differentiator_.Calculate(error);
121  // ローパスフィルタをかけない場合のD出力
122  pid_result_.before_lpf_d_term = pid_param_.kd * low_pass_differentiator_.GetLastRawOutput();
123 
124  // PID出力を計算
125  float before_clamp =
126  pid_result_.p_term + pid_result_.i_term + pid_result_.d_term + pid_result_.ff_term;
127  pid_result_.output = output_limit_.Clamp(before_clamp);
128

140  float Calculate(float error, float ff_output = 0.0f)
141  {
142  // 偏差を更新
143  calc_info_.error = error;
144  calc_info_.error_diff = low_pass_differentiator_.Calculate(error);
145  calc_info_.error_diff_before_lpf = low_pass_differentiator_.GetLastRawOutput();
146 
147  // フィードフォワード
148  calc_info_.ff_term = ff_output;
149 
150  // 比例
151  calc_info_.p_term = param_.kp * error;
152 
153  // 積分
154  // Anti-windup処理．出力値が飽和している場合は積分を行わない．
155  // 参考: https://www.youtube.com/watch?v=NVLXCwc8HzM&ab_channel=MATLAB
156  bool is_error_and_sum_same_sign = error * calc_info_.sum > 0.0f;
157  bool is_enabled_integrator = !(IsSaturated() && is_error_and_sum_same_sign);
158  if (is_enabled_integrator) {
159  calc_info_.i_term = param_.ki * integrator_.Calculate(error);
160  } else {
161  calc_info_.i_term = param_.ki * integrator_.Calculate(0);
162  }
163 
164  // 微分
165  // ローパスフィルタをかけたD出力
166  calc_info_.d_term = param_.kd * calc_info_.error_diff;
167 
168  // 各項の合計値
169  calc_info_.sum = calc_info_.p_term + calc_info_.i_term + calc_info_.d_term + calc_info_.ff_term;
170 
171  // 範囲内にクランプして出力値とする
172  calc_info_.output = output_limit_.Clamp(calc_info_.sum);
173 
174  return calc_info_.output;
175  }

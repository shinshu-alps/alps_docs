149 void DrawPixel(uint8_t x, uint8_t y, bool is_on = true)
150 {
151 if (x >= width_ || y >= height_) {
152 return;
153 }
154
155 const uint16_t index = x + (y / 8) * width_;
156 const uint8_t bit = 1 << (y % 8);
157
158 if (is_on) {
159 frame_buffer_[index] |= bit;
160 } else {
161 frame_buffer_[index] &= ~bit;
162 }
163 }

174 void DrawLine(uint8_t x0, uint8_t y0, uint8_t x1, uint8_t y1, bool is_on = true)
175 {
176 int dx = std::abs(x1 - x0), sx = x0 < x1 ? 1 : -1;
177 int dy = -std::abs(y1 - y0), sy = y0 < y1 ? 1 : -1;
178 int err = dx + dy, e2;
179
180 while (true) {
181 DrawPixel(x0, y0, is_on);
182 if (x0 == x1 && y0 == y1) {
183 break;
184 }
185 e2 = 2 * err;
186 if (e2 >= dy) {
187 err += dy;
188 x0 += sx;
189 }
190 if (e2 <= dx) {
191 err += dx;
192 y0 += sy;
193 }
194 }
195 }

274 void DrawCircle(uint8_t x0, uint8_t y0, uint8_t radius, bool is_on = true)
275 {
276 int x = radius;
277 int y = 0;
278 int err = 0;
279
280 while (x >= y) {
281 DrawPixel(x0 + x, y0 + y, is_on);
282 DrawPixel(x0 + y, y0 + x, is_on);
283 DrawPixel(x0 - y, y0 + x, is_on);
284 DrawPixel(x0 - x, y0 + y, is_on);
285 DrawPixel(x0 - x, y0 - y, is_on);
286 DrawPixel(x0 - y, y0 - x, is_on);
287 DrawPixel(x0 + y, y0 - x, is_on);
288 DrawPixel(x0 + x, y0 - y, is_on);
289
290 ++y;
291 if (err <= 0) {
292 err += 2 * y + 1;
293 } else {
294 --x;
295 err += 2 * (y - x + 1);
296 }
297 }
298 }

308 void FillCircle(uint8_t x0, uint8_t y0, uint8_t radius, bool is_on = true)
309 {
310 int x = radius;
311 int y = 0;
312 int err = 0;
313
314 while (x >= y) {
315 DrawHLine(x0 - x, y0 + y, 2 * x + 1, is_on);
316 DrawHLine(x0 - x, y0 - y, 2 * x + 1, is_on);
317 DrawHLine(x0 - y, y0 + x, 2 * y + 1, is_on);
318 DrawHLine(x0 - y, y0 - x, 2 * y + 1, is_on);
319
320 ++y;
321 if (err <= 0) {
322 err += 2 * y + 1;
323 } else {
324 --x;
325 err += 2 * (y - x + 1);
326 }
327 }

340 void DrawArc(
341 uint8_t x0,
342 uint8_t y0,
343 uint8_t radius,
344 float start_angle_deg,
345 float end_angle_deg,
346 bool is_on = true)
347 {
348 float start_rad = alps::cmn::util::DegToRad(start_angle_deg);
349 float end_rad = alps::cmn::util::DegToRad(end_angle_deg);
350 float angle_range = end_rad - start_rad;
351 const uint16_t steps = std::max<uint8_t>(1, static_cast<uint16_t>(radius * angle_range));
352
353 for (uint16_t i = 0; i <= steps; ++i) {
354 float theta = start_rad + (angle_range * i / steps);
355 uint8_t x = static_cast<uint8_t>(std::round(x0 + radius * std::cos(theta)));
356 uint8_t y = static_cast<uint8_t>(std::round(y0 + radius * std::sin(theta)));
357 DrawPixel(x, y, is_on);
358 }
359 }

369 void DrawChar(uint8_t x, uint8_t y, char c, bool is_on = true)
370 {
371 if (c < font_.first_char || c > font_.last_char) {
372 return;
373 }
374
375 // 文字のインデックス
376 size_t char_index = (c - font_.first_char) * font_.bytes_per_char;
377
378 // 必要なバイト数
379 uint8_t bytes_per_row = (font_.width + 7) / 8;
380
381 for (uint8_t row = 0; row < font_.height; ++row) {
382
383 // 各行のビット列の開始位置
384 const uint8_t *row_ptr = &font_.data[char_index + row * bytes_per_row];
385
386 for (uint8_t col = 0; col < font_.width; ++col) {
387
388 // 対象のビット位置
389 uint8_t byte_index = col / 8;
390 uint8_t bit_index = 7 - (col % 8); // MSB(左->右)
391
392 if (row_ptr[byte_index] & (1 << bit_index)) {
393 DrawPixel(x + col, y + row, is_on);
394 }
395 }
396 }

481 bool Show()
482 {
483 uint8_t col_start = 0;
484 uint8_t col_end = width_ - 1;
485 if (width_ != 128) {
486 uint8_t col_offset = (128 - width_) / 2;
487 col_start += col_offset;
488 col_end += col_offset;
489 }
490
491 // 列アドレスの範囲設定
492 if (!WriteCommand(kSetColAddr)) {
493 return false;
494 }
495
496 // 開始行
497 if (!WriteCommand(col_start)) {
498 return false;
499 }
500 // 終了行
501 if (!WriteCommand(col_end)) {
502 return false;
503 }
504
505 // ページアドレスの範囲設定
506 if (!WriteCommand(kSetPageAddr)) {
507 return false;
508 }
509 // 開始ページ
510 if (!WriteCommand(0x00)) {
511 return false;
512 }
513 // 終了ページ
514 if (!WriteCommand(static_cast<Command>((height_ / 8) - 1))) {
515 return false;
516 }
517
518 return WriteData(frame_buffer_);
519 }

542 bool WriteData(etl::span<const Data> data)
543 {
544 std::vector<uint8_t> buf;
545 buf.reserve(1 + data.size());
546 buf.push_back(kControlData);
547 buf.insert(buf.end(), data.begin(), data.end());
548 return i2c_io_.CallWrite(i2c_address_, buf);

642  uint16_t DecodeTimeout(uint16_t value)
643  {
644  // format: "(LSByte * 2^MSByte) + 1"
645  return (uint16_t)((value & 0x00FF) << (uint16_t)((value & 0xFF00) >> 8)) + 1;

654  uint16_t EncodeTimeout(uint32_t timeout_mclks)
655  {
656  // format: "(LSByte * 2^MSByte) + 1"
657 
658  uint32_t ls_byte = 0;
659  uint16_t ms_byte = 0;
660 
661  if (timeout_mclks > 0) {
662  ls_byte = timeout_mclks - 1;
663 
664  while ((ls_byte & 0xFFFFFF00) > 0) {
665  ls_byte >>= 1;
666  ms_byte++;
667  }
668 
669  return (ms_byte << 8) | (ls_byte & 0xFF);
670  } else {
671  return 0;
672  }
673  }

764  bool SetVcselPulsePeriod(VcselPeriodType type, uint8_t period_pclks)
765  {
766  uint8_t vcsel_period_reg = EncodeVcselPeriod(period_pclks);
767 
768  SequenceStepEnables enables;
769  SequenceStepTimeouts timeouts;
770 
771  GetSequenceStepEnables(&enables);
772  GetSequenceStepTimeouts(&enables, &timeouts);
773 
774  // "Apply specific settings for the requested clock period"
775  // "Re-calculate and apply timeouts, in macro periods"
776 
777  // "When the VCSEL period for the pre or final range is changed,
778  // the corresponding timeout must be read from the device using
779  // the current VCSEL period, then the new VCSEL period can be
780  // applied. The timeout then must be written back to the device
781  // using the new VCSEL period.
782  //
783  // For the MSRC timeout, the same applies - this timeout being
784  // dependant on the pre-range vcsel period."
785 
786  if (type == kVcselPeriodPreRange) {
787  // "Set phase check limits"
788  switch (period_pclks) {
789  case 12:
790  WriteReg(kPreRangeConfigValidPhaseHigh, 0x18);
791  break;
792 
793  case 14:
794  WriteReg(kPreRangeConfigValidPhaseHigh, 0x30);
795  break;
796 
797  case 16:
798  WriteReg(kPreRangeConfigValidPhaseHigh, 0x40);
799  break;
800 
801  case 18:
802  WriteReg(kPreRangeConfigValidPhaseHigh, 0x50);
803  break;
804 
805  default:
806  // invalid period
807  return false;
808  }
809  WriteReg(kPreRangeConfigValidPhaseLow, 0x08);
810 
811  // apply new VCSEL period
812  WriteReg(kPreRangeConfigVcselPeriod, vcsel_period_reg);
813 
814  // update timeouts
815 
816  // set_sequence_step_timeout() begin
817  // (SequenceStepId == VL53L0X_SEQUENCESTEP_PRE_RANGE)
818 
819  uint16_t new_pre_range_timeout_mclks =
820  TimeoutMicrosecondsToMclks(timeouts.pre_range_us, period_pclks);
821 
822  WriteReg16Bit(kPreRangeConfigTimeoutMacropHi, EncodeTimeout(new_pre_range_timeout_mclks));
823 
824  // set_sequence_step_timeout() end
825 
826  // set_sequence_step_timeout() begin
827  // (SequenceStepId == VL53L0X_SEQUENCESTEP_MSRC)
828 
829  uint16_t new_msrc_timeout_mclks =
830  TimeoutMicrosecondsToMclks(timeouts.msrc_dss_tcc_us, period_pclks);
831 
832  WriteReg(
833  kMsrcConfigTimeoutMacrop,
834  (new_msrc_timeout_mclks > 256) ? 255 : (new_msrc_timeout_mclks - 1));
835 
836  // set_sequence_step_timeout() end
837  } else if (type == kVcselPeriodFinalRange) {
838  switch (period_pclks) {
839  case 8:
840  WriteReg(kFinalRangeConfigValidPhaseHigh, 0x10);
841  WriteReg(kFinalRangeConfigValidPhaseLow, 0x08);
842  WriteReg(kGlobalConfigVcselWidth, 0x02);
843  WriteReg(kAlgoPhasecalConfigTimeout, 0x0C);
844  WriteReg(0xFF, 0x01);
845  WriteReg(kAlgoPhasecalLim, 0x30);
846  WriteReg(0xFF, 0x00);
847  break;
848 
849  case 10:
850  WriteReg(kFinalRangeConfigValidPhaseHigh, 0x28);
851  WriteReg(kFinalRangeConfigValidPhaseLow, 0x08);
852  WriteReg(kGlobalConfigVcselWidth, 0x03);
853  WriteReg(kAlgoPhasecalConfigTimeout, 0x09);
854  WriteReg(0xFF, 0x01);
855  WriteReg(kAlgoPhasecalLim, 0x20);
856  WriteReg(0xFF, 0x00);
857  break;
858 
859  case 12:
860  WriteReg(kFinalRangeConfigValidPhaseHigh, 0x38);
861  WriteReg(kFinalRangeConfigValidPhaseLow, 0x08);
862  WriteReg(kGlobalConfigVcselWidth, 0x03);
863  WriteReg(kAlgoPhasecalConfigTimeout, 0x08);
864  WriteReg(0xFF, 0x01);
865  WriteReg(kAlgoPhasecalLim, 0x20);
866  WriteReg(0xFF, 0x00);
867  break;
868 
869  case 14:
870  WriteReg(kFinalRangeConfigValidPhaseHigh, 0x48);
871  WriteReg(kFinalRangeConfigValidPhaseLow, 0x08);
872  WriteReg(kGlobalConfigVcselWidth, 0x03);
873  WriteReg(kAlgoPhasecalConfigTimeout, 0x07);
874  WriteReg(0xFF, 0x01);
875  WriteReg(kAlgoPhasecalLim, 0x20);
876  WriteReg(0xFF, 0x00);
877  break;
878 
879  default:
880  // invalid period
881  return false;
882  }
883 
884  // apply new VCSEL period
885  WriteReg(kFinalRangeConfigVcselPeriod, vcsel_period_reg);
886 
887  // update timeouts
888 
889  // set_sequence_step_timeout() begin
890  // (SequenceStepId == VL53L0X_SEQUENCESTEP_FINAL_RANGE)
891 
892  // "For the final range timeout, the pre-range timeout
893  // must be added. To do this both final and pre-range
894  // timeouts must be expressed in macro periods MClks
895  // because they have different vcsel periods."
896 
897  uint16_t new_final_range_timeout_mclks =
898  TimeoutMicrosecondsToMclks(timeouts.final_range_us, period_pclks);
899 
900  if (enables.pre_range) {
901  new_final_range_timeout_mclks += timeouts.pre_range_mclks;
902  }
903 
904  WriteReg16Bit(kFinalRangeConfigTimeoutMacropHi, EncodeTimeout(new_final_range_timeout_mclks));
905 
906  // set_sequence_step_timeout end
907  } else {
908  // invalid type
909  return false;
910  }
911 
912  // "Finally, the timing budget must be re-applied"
913 
914  SetMeasurementTimingBudget(measurement_timing_budget_us_);
915 
916  // "Perform the phase calibration. This is needed after changing on vcsel period."
917  // VL53L0X_perform_phase_calibration() begin
918 
919  uint8_t sequence_config = ReadReg(kSystemSequenceConfig);
920  WriteReg(kSystemSequenceConfig, 0x02);
921  PerformSingleRefCalibration(0x0);
922  WriteReg(kSystemSequenceConfig, sequence_config);
923 
924  // VL53L0X_perform_phase_calibration() end
925

988  bool SetMeasurementTimingBudget(uint32_t budget_us)
989  {
990  SequenceStepEnables enables;
991  SequenceStepTimeouts timeouts;
992 
993  uint16_t const kStartOverhead = 1910;
994  uint16_t const kEndOverhead = 960;
995  uint16_t const kMsrcOverhead = 660;
996  uint16_t const kTccOverhead = 590;
997  uint16_t const kDssOverhead = 690;
998  uint16_t const kPreRangeOverhead = 660;
999  uint16_t const kFinalRangeOverhead = 550;
1000 
1001  uint32_t used_budget_us = kStartOverhead + kEndOverhead;
1002 
1003  GetSequenceStepEnables(&enables);
1004  GetSequenceStepTimeouts(&enables, &timeouts);
1005 
1006  if (enables.tcc) {
1007  used_budget_us += (timeouts.msrc_dss_tcc_us + kTccOverhead);
1008  }
1009 
1010  if (enables.dss) {
1011  used_budget_us += 2 * (timeouts.msrc_dss_tcc_us + kDssOverhead);
1012  } else if (enables.msrc) {
1013  used_budget_us += (timeouts.msrc_dss_tcc_us + kMsrcOverhead);
1014  }
1015 
1016  if (enables.pre_range) {
1017  used_budget_us += (timeouts.pre_range_us + kPreRangeOverhead);
1018  }
1019 
1020  if (enables.final_range) {
1021  used_budget_us += kFinalRangeOverhead;
1022 
1023  // "Note that the final range timeout is determined by the timing
1024  // budget and the sum of all other timeouts within the sequence.
1025  // If there is no room for the final range timeout, then an error
1026  // will be set. Otherwise the remaining time will be applied to
1027  // the final range."
1028 
1029  if (used_budget_us > budget_us) {
1030  // "Requested timeout too big."
1031  return false;
1032  }
1033 
1034  uint32_t final_range_timeout_us = budget_us - used_budget_us;
1035 
1036  // set_sequence_step_timeout() begin
1037  // (SequenceStepId == VL53L0X_SEQUENCESTEP_FINAL_RANGE)
1038 
1039  // "For the final range timeout, the pre-range timeout
1040  // must be added. To do this both final and pre-range
1041  // timeouts must be expressed in macro periods MClks
1042  // because they have different vcsel periods."
1043 
1044  uint32_t final_range_timeout_mclks =
1045  TimeoutMicrosecondsToMclks(final_range_timeout_us, timeouts.final_range_vcsel_period_pclks);
1046 
1047  if (enables.pre_range) {
1048  final_range_timeout_mclks += timeouts.pre_range_mclks;
1049  }
1050 
1051  WriteReg16Bit(kFinalRangeConfigTimeoutMacropHi, EncodeTimeout(final_range_timeout_mclks));
1052 
1053  // set_sequence_step_timeout() end
1054 
1055  measurement_timing_budget_us_ = budget_us; // store for internal reuse
1056  }

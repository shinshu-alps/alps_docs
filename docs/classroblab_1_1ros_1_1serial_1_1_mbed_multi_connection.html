詳解

パソコンに接続されている複数のMbedとシリアル通信を行う

Linux(Ubuntu)環境下で接続されている複数のMbedとシリアル通信を行う 切断や再接続にも対応

注釈
複数のノードでの使用は考慮されていない

例

constexpr float kCtrlPeriod{0.02f};
float g_encoder_deg;
void OnReceiveEncoder(float &&enc)
{
g_encoder_deg = enc;
}
int main(int argc, char **argv)
{
ros::init(argc, argv, "test_os6_lib");
ros::NodeHandle nh;
ros::Rate loop_rate(1.0 / kCtrlPeriod);
roblab::ros::MbedMultiConnection mbeds{B115200};
auto mbed0_connection = mbeds.GetPacketIdManager("mbed0");
auto motor_pwm = mbed0_connection->CreateSender<float>(0);
mbed0_connection->RegisterReceiveCallback<float>(0, OnReceiveEncoder);
roblab::common::Pid pid{kCtrlPeriod};
while (ros::ok())
{
motor_pwm.EnqueueSendData(12.345);
mbeds.Update();
ros::spinOnce();
loop_rate.sleep();
}
return 0;
}

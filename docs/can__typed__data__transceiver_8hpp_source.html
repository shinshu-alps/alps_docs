102template <class T>
103class CanPublisher
104{
105public:
114 template <class CanIo, size_t kMaxSubscriberNum>
115 CanPublisher(CanTypedDataTransceiver<CanIo, kMaxSubscriberNum> & transceiver, CanDataFrame::Id id)
116 : id_{id}
117 {
118 // CAN送信関数を取得
119 auto & can_io = transceiver.GetCanIoRef();
120 can_write_func_ = etl::delegate<bool(
121 const CanDataFrame &)>::template create<ICanIo<CanIo>, &ICanIo<CanIo>::CallTryWrite>(can_io);
122 }
123
129 void Publish(const T & data)
130 {
131 // 書き込み方法がない場合は何もしない
132 if (!can_write_func_.is_valid()) {
133 ALPS_LOG_WARN("CanPublisher has no way to write CAN data.");
134 return;
135 }
136
137 etl::circular_buffer<uint8_t, RequiredBufferSizeForCanTypedDataCommunication<T>()>
138 send_data_bytes;
139 CanDataFrame data_frame;
140 data_frame.id = id_;
141
142 Serializer<T>::Serialize(data, send_data_bytes);
143
144 // if constexpr
145 if (
146 Serializer<T>::RequiredByteSize() >
147 CanDataFrame::Data::MAX_SIZE) // 1つのメッセージで収まらない場合
148 {
149 alps::cmn::communication::Cobs::Encode(send_data_bytes); // COBS変換
150
151 for (size_t i = 0; i < send_data_bytes.size(); ++i) {
152 data_frame.data.push_back(send_data_bytes[i]);
153
154 if (data_frame.data.size() == CanDataFrame::Data::MAX_SIZE) // バッファが一杯になったら送信
155 {
156 can_write_func_(data_frame);
157 data_frame.data.clear();
158 }
159 }
160
161 // 残りのバッファを送信
162 if (!data_frame.data.empty()) {
163 can_write_func_(data_frame);
164 }
165 } else // 1つのメッセージで収まる場合
166 {
167 for (size_t i = 0; i < send_data_bytes.size(); ++i) {
168 data_frame.data.push_back(send_data_bytes[i]);
169 }
170 can_write_func_(data_frame);
171 };
172 }
173
175 const CanDataFrame::Id id_;
176
177private:
179 etl::delegate<bool(const CanDataFrame &)> can_write_func_;
180};

187template <class T>
188class CanSubscriber
189{
190public:
201 template <class CanIo, size_t kMaxSubscriberNum>
202 CanSubscriber(
203 CanTypedDataTransceiver<CanIo, kMaxSubscriberNum> & transceiver,
204 CanDataFrame::Id id,
205 etl::delegate<void(const T &)> receive_data_callback)
206 : id_{id}, receive_data_callback_{receive_data_callback}
207 {
208 // 受信コールバック関数を登録
209 transceiver.GetCallbackHandlerRef().RegisterReceiveCallback(
210 id_,
211 etl::delegate<void(const CanDataFrame::Data &)>::
212 create<CanSubscriber<T>, &CanSubscriber<T>::OnSubscribeCanFrame>(*this));
213 }
214
220 void SetReceiveDataCallback(etl::delegate<void(const T &)> receive_data_callback)
221 {
222 receive_data_callback_ = receive_data_callback;
223 }
224
226 const CanDataFrame::Id id_;
227
228private:
234 void OnSubscribeCanFrame(const CanDataFrame::Data & data)
235 {
236 // 受信したデータをバッファに追加
237 for (size_t i = 0; i < data.size(); ++i) {
238 received_bytes_.push(data[i]);
239 }
240
241 // if constexpr
242 if (
243 Serializer<T>::RequiredByteSize() >
244 CanDataFrame::Data::MAX_SIZE) // 1つのメッセージで収まらない場合
245 {
246 // COBS変換の終端を受信したらCobsバイト列をデコードしてコールバックを呼ぶ
247 if (received_bytes_.back() == Cobs::kEndVal) {
248 T data;
249 etl::circular_buffer<uint8_t, Serializer<T>::RequiredByteSize()> decoded_bytes;
250 if (alps::cmn::communication::Cobs::TryDecode(received_bytes_, decoded_bytes)) {
251 bool result = Serializer<T>::Deserialize(decoded_bytes, data);
252 if (result) {
253 receive_data_callback_(data);
254 }
255 } else {
256 ALPS_LOG_WARN(
257 "Subscribed CAN data(CAN-ID: %x) is broken. COBS decode failed.", this->id_);
258 }
259 }
260 } else // 1つのメッセージで収まる場合
261 {
262 T data;
263 bool result = Serializer<T>::Deserialize(received_bytes_, data);
264 if (result) {
265 receive_data_callback_(data);
266 }
267 }
268 }
269
271 etl::delegate<void(const T &)> receive_data_callback_;
273 etl::circular_buffer<uint8_t, RequiredBufferSizeForCanTypedDataCommunication<T>()>
274 received_bytes_;
275};

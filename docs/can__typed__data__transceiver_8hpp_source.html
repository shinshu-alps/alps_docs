81 if (!has_new_data_) {

83 }
84
85 etl::circular_buffer<uint8_t, RequiredBufferSizeForCanTypedDataCommunication<T>()>
86 send_data_bytes;
87 CanDataFrame data_frame;
88 data_frame.id = id_;
89
90 DataSerializer<T>::Serialize(data_, send_data_bytes);
91
92 // if constexpr
93 if (
94 SerializeTypeInfo<T>::RequiredBytesSize() >
95 CanDataFrame::Data::MAX_SIZE) // 1つのメッセージで収まらない場合
96 {
97 alps::cmn::communication::Cobs::Encode(send_data_bytes); // COBS変換
98
99 for (size_t i = 0; i < send_data_bytes.size(); ++i) {
100 data_frame.data.push_back(send_data_bytes[i]);
101
102 if (data_frame.data.size() == CanDataFrame::Data::MAX_SIZE) // バッファが一杯になったら送信
103 {
104 if (!can_io.CallTryWrite(data_frame)) {
105 return false;
106 }
107 data_frame.data.clear();
108 }
109 }
110 } else // 1つのメッセージで収まる場合
111 {
112 for (size_t i = 0; i < send_data_bytes.size(); ++i) {
113 data_frame.data.push_back(send_data_bytes[i]);
114 }
115 };
116
117 // データを送信
118 if (!can_io.CallTryWrite(data_frame)) {
119 return false;
120 }
121
122 has_new_data_ = false;
123 return true;
124 }

163 void OnSubscribeCanFrame(const CanDataFrame::Data & data)
164 {
165 // 受信したデータをバッファに追加
166 for (size_t i = 0; i < data.size(); ++i) {
167 received_bytes_.push(data[i]);
168 }
169
170 // if constexpr
171 if (
172 SerializeTypeInfo<T>::RequiredBytesSize() >
173 CanDataFrame::Data::MAX_SIZE) // 1つのメッセージで収まらない場合
174 {
175 if (
176 received_bytes_.back() ==
177 Cobs::kEndVal) // COBS変換の終端を受信したらCobsバイト列をデコードしてコールバックを呼ぶ
178 {
179 T data;
180 etl::circular_buffer<uint8_t, SerializeTypeInfo<T>::RequiredBytesSize()> decoded_bytes;
181 if (alps::cmn::communication::Cobs::TryDecode(received_bytes_, decoded_bytes)) {
182 bool result = DataSerializer<T>::Deserialize(decoded_bytes, data);
183 if (result) {
184 receive_data_callback_(data);
185 }
186 }
187 }
188 } else // 1つのメッセージで収まる場合
189 {
190 T data;
191 bool result = DataSerializer<T>::Deserialize(received_bytes_, data);
192 if (result) {
193 receive_data_callback_(data);
194 }
195 }
196 }
197
198public:
200 const CanDataFrame::Id id_;
201
202private:
204 etl::delegate<void(const T &)> receive_data_callback_;
206 etl::circular_buffer<uint8_t, RequiredBufferSizeForCanTypedDataCommunication<T>()>
207 received_bytes_;

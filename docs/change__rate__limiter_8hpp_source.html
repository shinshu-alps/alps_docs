108  // 変化率を計算
109  auto current_timestamp = Clock::now();
110  auto delta_time =
111  std::chrono::duration_cast<time::Seconds>(current_timestamp - last_timestamp_);
112  auto change_rate = (input - last_output_) / delta_time.count();
113 
114  // 変化率が閾値を超えていたら、閾値に制限する
115  float output = input;
116  if (change_rate > max_change_rate_) {
117  output = last_output_ + max_change_rate_ * delta_time.count();
118  } else if (change_rate < lower_change_rate_) {
119  output = last_output_ + lower_change_rate_ * delta_time.count();
120  }
121 
122  // 次回のために値を更新
123  last_timestamp_ = current_timestamp;
124  last_output_ = output;
125

127  }

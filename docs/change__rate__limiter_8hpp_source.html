92 // 初回の値はそのまま出力する
93 if (first_flag_) {
94 first_flag_ = false;
95 last_timestamp_ = Clock::now();
96 last_output_ = input;
97 return input;
98 }
99
100 // 変化率を計算
101 auto current_timestamp = Clock::now();
102 auto delta_time =
103 std::chrono::duration_cast<time::Seconds>(current_timestamp - last_timestamp_);
104 auto change_rate = (input - last_output_) / delta_time.count();
105
106 // 時間経過が0の場合は前回の値をそのまま返す
107 if (delta_time.count() == 0) {
108 return last_output_;
109 }
110
111 // 変化率が閾値を超えていたら、閾値に制限する
112 float output = input;
113 if (change_rate > change_rate_limit_.max) {
114 output = last_output_ + change_rate_limit_.max * delta_time.count();
115 } else if (change_rate < change_rate_limit_.lowest) {
116 output = last_output_ + change_rate_limit_.lowest * delta_time.count();
117 }
118
119 // 次回のために値を更新
120 last_timestamp_ = current_timestamp;
121 last_output_ = output;
122
123 return output;
124 }

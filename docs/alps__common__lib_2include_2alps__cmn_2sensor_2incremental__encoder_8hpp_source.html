120 float GetVelocity()
121 {
122 using namespace std::chrono_literals;
123 using Seconds = std::chrono::duration<float, std::ratio<1>>;
124 auto timestamp_now = Clock::now();
125 constexpr Seconds kMinElapsed{1ms};
126 auto elapsed =
127 std::chrono::duration_cast<Seconds>(timestamp_now - last_calc_velocity_time_point_);
128
129 if (elapsed < kMinElapsed) {
130 return velocity_;
131 }
132
133 double now_angle = GetAngle();
134 velocity_ = (now_angle - last_calc_velocity_angle_) / elapsed.count();
135
136 last_calc_velocity_time_point_ = timestamp_now;
137 last_calc_velocity_angle_ = now_angle;
138 return velocity_;
139 }

158 void IncrementPulses(bool a_state)
159 {
160 int8_t b_state = read_B_.CallRead();
161 curr_state_ = (static_cast<int8_t>(a_state) << 1) | (b_state);
162
163 if ((prev_state_ == 0x3 && curr_state_ == 0x0) || (prev_state_ == 0x0 && curr_state_ == 0x3)) {
164 ++pulse_change_count_;
165 } else if (
166 (prev_state_ == 0x2 && curr_state_ == 0x1) || (prev_state_ == 0x1 && curr_state_ == 0x2)) {
167 --pulse_change_count_;
168 }
169
170 prev_state_ = curr_state_;
171 }

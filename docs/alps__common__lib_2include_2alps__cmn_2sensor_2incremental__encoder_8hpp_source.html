145 void IncrementPulses(bool a_state)
146 {
147 int8_t b_state = read_B_.CallRead();
148 curr_state_ = (static_cast<int8_t>(a_state) << 1) | (b_state);
149
150 if ((prev_state_ == 0x3 && curr_state_ == 0x0) || (prev_state_ == 0x0 && curr_state_ == 0x3)) {
151 ++pulse_change_count_;
152 } else if (
153 (prev_state_ == 0x2 && curr_state_ == 0x1) || (prev_state_ == 0x1 && curr_state_ == 0x2)) {
154 --pulse_change_count_;
155 }
156
157 prev_state_ = curr_state_;
158 }

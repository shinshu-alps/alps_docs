103 bool GetRgbColor(RgbColor & output)

106 if (!ReadMultiRegister(kClearDataLowReg | kCommandBit, reg_values)) {

110 output.clear = (reg_values[1] << 8) | reg_values[0]; // クリア値
111 output.red = (reg_values[3] << 8) | reg_values[2]; // 赤色
112 output.green = (reg_values[5] << 8) | reg_values[4]; // 緑色
113 output.blue = (reg_values[7] << 8) | reg_values[6]; // 青色

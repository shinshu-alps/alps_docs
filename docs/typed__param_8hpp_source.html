183  void OnParameterEvent(const rcl_interfaces::msg::ParameterEvent::SharedPtr event)
184  {
185  // サーバーノード名が一致しない場合は何もしない
186  if (event->node.substr(1) != server_node_name_) { // 1文字目の'/'を除いて比較
187  return;
188  }
189 
190  // 削除パラメータを監視
191  for (const auto & param : event->deleted_parameters) {
192  // このパラメータを構成するパラメータである場合
193  if (param_element_names_.count(param.name) > 0) {
194  current_value_ = std::nullopt; // 値を未セット状態に戻す
195  param_map_.RemoveParam(param.name); // マップから削除
196  }
197  }
198 
199  // 新規パラメータと変更パラメータを監視
200  bool is_this_param_changed = false;
201  for (const auto & target_params : {event->new_parameters, event->changed_parameters}) {
202  for (const auto & param : target_params) {
203  // このパラメータを構成するパラメータである場合
204  if (param_element_names_.count(param.name) > 0) {
205  is_this_param_changed = true;
206  param_map_.SetParam(param.name, rclcpp::Parameter::from_parameter_msg(param));
207  }
208  }
209  }
210 
211  // 関係ないパラメータの変更時は何もしない
212  if (!is_this_param_changed) {
213  return;
214  }
215 
216  // 値の更新
217  UpdateValue();
218 
219  if (HasValidValue() == false) {
220  if (std::all_of(
221  param_element_names_.begin(),
222  param_element_names_.end(),
223  [this](const std::string & name) { return param_map_.Contains(name); })) {
224  // 全てのパラメータを持っているのにも関わらず値取得に失敗した場合エラー
225  RCLCPP_WARN(logger_, "Failed to get parameter. Elements may be invalid.");
226  RCLCPP_WARN(
227  this->logger_,
228  "\tserver_node_name: %s, param_name: %s",
229  this->GetServerNodeName().data(),
230  this->GetParamName().data());
231  }
232  }
233 
234  // コールバック関数の呼び出し
235  InvokeOnChangeCallback();
236  }

347  void CbTimerTryGetInitialValue()
348  {
349  // サービスが準備できていない場合は何もしない
350  if (!this->IsServerReady()) {
351  return;
352  }
353  // 既に有効値を持っている場合は何もしない
354  if (this->HasValidValue()) {
355  // タイマーの停止
356  this->timer_try_get_initial_value_->cancel();
357  return;
358  }
359 
360  std::vector<std::string> param_element_names{
361  this->GetParamElementNames().begin(), this->GetParamElementNames().end()};
362 
363  this->initial_param_request_result_ = this->parameters_client_->get_parameters(
364  param_element_names, [this](std::shared_future<std::vector<rclcpp::Parameter>> result) {
365  // 受け取れなかった場合は何もしない
366  if (!result.valid()) {
367  return;
368  }
369 
370  // タイマーの停止
371  this->timer_try_get_initial_value_->cancel();
372 
373  // 既に有効値を持っている場合は何もしない
374  if (this->HasValidValue()) {
375  return;
376  }
377 
378  // 初期値取得
379  for (const auto & param : result.get()) {
380  this->param_map_.SetParam(param.get_name(), param);
381  }
382  this->UpdateValue();
383 
384  // コールバック関数の呼び出し
385  this->InvokeOnChangeCallback();
386 
387  // 取得したパラメータから値を取得できなかった場合警告
388  if (this->HasValidValue() == false) {
389  RCLCPP_WARN(
390  this->logger_,
391  "Failed to get initial parameter. Param type or element may be invalid. "
392  "server_node_name: "
393  "%s, param_name: %s",
394  this->GetServerNodeName().data(),
395  this->GetParamName().data());
396  }
397  });
398  }

401  void CbTimerCheckServer()
402  {
403  // タイマーの停止
404  timer_check_server_->cancel();
405  timer_try_get_initial_value_->cancel();
406 
407  // まだサーバーが準備できていない場合警告
408  if (this->IsServerReady() == false) {
409  RCLCPP_WARN(
410  this->logger_,
411  "Parameter server is not found. server_node_name: %s, param_name: %s",
412  this->GetServerNodeName().data(),
413  this->GetParamName().data());
414  }
415  }

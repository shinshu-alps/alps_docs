152  void OnParameterEvent(const rcl_interfaces::msg::ParameterEvent::SharedPtr event)
153  {
154  // サーバーノード名が一致しない場合は何もしない
155  if (event->node.substr(1) != server_node_name_) { // 1文字目の'/'を除いて比較
156  return;
157  }
158 
159  // 削除パラメータを監視
160  for (const auto & param : event->deleted_parameters) {
161  // このパラメータを構成するパラメータである場合
162  if (param_element_names_.count(param.name) > 0) {
163  current_value_ = std::nullopt; // 値を未セット状態に戻す
164  param_map_.RemoveParam(param.name); // マップから削除
165  }
166  }
167 
168  // 新規パラメータと変更パラメータを監視
169  bool is_this_param_changed = false;
170  for (const auto & target_params : {event->new_parameters, event->changed_parameters}) {
171  for (const auto & param : target_params) {
172  // このパラメータを構成するパラメータである場合
173  if (param_element_names_.count(param.name) > 0) {
174  is_this_param_changed = true;
175  param_map_.SetParam(param.name, rclcpp::Parameter::from_parameter_msg(param));
176  }
177  }
178  }
179 
180  // 関係ないパラメータの変更時は何もしない
181  if (!is_this_param_changed) {
182  return;
183  }
184 
185  // 値の更新
186  current_value_ = std::nullopt;
187  current_value_ = GetParam();
188 
189  if (current_value_) {
190  // コールバック関数の呼び出し
191  if (callback_) {
192  callback_(*current_value_);
193  }
194  } else {
195  if (std::all_of(
196  param_element_names_.begin(),
197  param_element_names_.end(),
198  [this](const std::string & name) { return param_map_.Contains(name); })) {
199  // 全てのパラメータを持っているのにも関わらず値取得に失敗した場合エラー
200  RCLCPP_WARN(logger_, "Failed to get parameter. Elements may be invalid.");
201  RCLCPP_WARN(
202  this->logger_,
203  "\tserver_node_name: %s, param_name: %s",
204  this->GetServerNodeName().data(),
205  this->GetParamName().data());
206  }
207  }
208  }

177  void OnParameterEvent(const rcl_interfaces::msg::ParameterEvent::SharedPtr event)
178  {
179  // サーバーノード名が一致しない場合は何もしない
180  if (event->node.substr(1) != server_node_name_) { // 1文字目の'/'を除いて比較
181  return;
182  }
183 
184  // 削除パラメータを監視
185  for (const auto & param : event->deleted_parameters) {
186  // このパラメータを構成するパラメータである場合
187  if (param_element_names_.count(param.name) > 0) {
188  current_value_ = std::nullopt; // 値を未セット状態に戻す
189  param_map_.RemoveParam(param.name); // マップから削除
190  }
191  }
192 
193  // 新規パラメータと変更パラメータを監視
194  bool is_this_param_changed = false;
195  for (const auto & target_params : {event->new_parameters, event->changed_parameters}) {
196  for (const auto & param : target_params) {
197  // このパラメータを構成するパラメータである場合
198  if (param_element_names_.count(param.name) > 0) {
199  is_this_param_changed = true;
200  param_map_.SetParam(param.name, rclcpp::Parameter::from_parameter_msg(param));
201  }
202  }
203  }
204 
205  // 関係ないパラメータの変更時は何もしない
206  if (!is_this_param_changed) {
207  return;
208  }
209 
210  // 値の更新
211  UpdateValue();
212 
213  if (HasValidValue() == false) {
214  if (std::all_of(
215  param_element_names_.begin(),
216  param_element_names_.end(),
217  [this](const std::string & name) { return param_map_.Contains(name); })) {
218  // 全てのパラメータを持っているのにも関わらず値取得に失敗した場合エラー
219  RCLCPP_WARN(logger_, "Failed to get parameter. Elements may be invalid.");
220  RCLCPP_WARN(
221  this->logger_,
222  "\tserver_node_name: %s, param_name: %s",
223  this->GetServerNodeName().data(),
224  this->GetParamName().data());
225  }
226  }
227 
228  // コールバック関数の呼び出し
229  InvokeOnChangeCallback();
230  }

280  }

285  // コールバック関数の呼び出し

321  {

323 
324  // 初期値取得タイマーの開始
325  this->timer_try_get_initial_value_ =
326  node.create_wall_timer(500ms, std::bind(&TypedParamClient::CbTimerTryGetInitialValue, this));
327 
328  // サーバーチェックタイマーの開始
329  this->timer_check_server_ =
330  node.create_wall_timer(3s, std::bind(&TypedParamClient::CbTimerCheckServer, this));
331  }
332 
333 private:
335  void CbTimerTryGetInitialValue()
336  {
337  // サービスが準備できていない場合は何もしない
338  if (!this->IsServerReady()) {
339  return;
340  }
341  // 既に有効値を持っている場合は何もしない
342  if (this->HasValidValue()) {
343  // タイマーの停止
344  this->timer_try_get_initial_value_->cancel();
345  return;
346  }
347 
348  std::vector<std::string> param_element_names{
349  this->GetParamElementNames().begin(), this->GetParamElementNames().end()};
350 
351  this->initial_param_request_result_ = this->parameters_client_->get_parameters(
352  param_element_names, [this](std::shared_future<std::vector<rclcpp::Parameter>> result) {
353  // 受け取れなかった場合は何もしない
354  if (!result.valid()) {
355  return;
356  }
357 
358  // タイマーの停止
359  this->timer_try_get_initial_value_->cancel();
360 
361  // 既に有効値を持っている場合は何もしない
362  if (this->HasValidValue()) {
363  return;
364  }
365 
366  // 初期値取得
367  for (const auto & param : result.get()) {
368  this->param_map_.SetParam(param.get_name(), param);
369  }
370  this->UpdateValue();
371 
372  // コールバック関数の呼び出し
373  this->InvokeOnChangeCallback();
374 
375  // 取得したパラメータから値を取得できなかった場合警告
376  if (this->HasValidValue() == false) {
377  RCLCPP_WARN(
378  this->logger_,
379  "Failed to get initial parameter. Param type or element may be invalid. "
380  "server_node_name: "
381  "%s, param_name: %s",
382  this->GetServerNodeName().data(),
383  this->GetParamName().data());
384  }
385  });
386  }
387 
389  void CbTimerCheckServer()
390  {
391  // タイマーの停止
392  timer_check_server_->cancel();
393  timer_try_get_initial_value_->cancel();
394 
395  // まだサーバーが準備できていない場合警告
396  if (this->IsServerReady() == false) {
397  RCLCPP_WARN(
398  this->logger_,
399  "Parameter server is not found. server_node_name: %s, param_name: %s",
400  this->GetServerNodeName().data(),
401  this->GetParamName().data());
402  }
403  }

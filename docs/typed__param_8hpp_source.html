182  void OnParameterEvent(const rcl_interfaces::msg::ParameterEvent::SharedPtr event)
183  {
184  // サーバーノード名が一致しない場合は何もしない
185  if (event->node.substr(1) != server_node_name_) { // 1文字目の'/'を除いて比較
186  return;

188 
189  // 削除パラメータを監視
190  for (const auto & param : event->deleted_parameters) {
191  if (element_name_to_is_set_.count(param.name) != 0) {
192  element_name_to_is_set_[param.name] = false; // 値を未セット状態に戻す
193  }
194  }
195 
196  // 新規パラメータと変更パラメータを監視
197  bool is_this_param_changed = false;
198  for (const auto & target_params : {event->new_parameters, event->changed_parameters}) {
199  for (const auto & param : target_params) {
200  if (element_name_to_is_set_.count(param.name) != 0) {
201  element_name_to_is_set_[param.name] = true;
202  is_this_param_changed = true;
203  param_map_.SetParam(param.name, rclcpp::Parameter::from_parameter_msg(param));
204  }
205  }
206  }
207 
208  // 関係ないパラメータの変更時は何もしない
209  if (!is_this_param_changed) {
210  return;
211  }
212 
213  // パラメータが構築されていない場合は何もしない
214  if (IsServerReady() == false) {
215  return;
216  }
217

220  if (value) {

222  callback_(*value);
223  }
224  } else {

226  RCLCPP_WARN(
227  node_.get_logger(),
228  "\tserver_node_name: %s, param_name: %s",
229  server_node_name_.c_str(),
230  param_name_.c_str());
231  }
